15class Resampler(): 

16 """Abstract base class for implementing resamplers. 

17 

18 All subclasses of Resampler must implement the 'resample' method. 

19 

20 If extra reporting on resampling and resampler updates desired 

21 subclassed resamplers should update the following class constants 

22 for specifying the decision class and the names, shapes, data 

23 types, and table-like records for each record group: 

24 

25 - DECISION 

26 - RESAMPLING_FIELDS 

27 - RESAMPLING_SHAPES 

28 - RESAMPLING_DTYPES 

29 - RESAMPLING_RECORD_FIELDS 

30 - RESAMPLER_FIELDS 

31 - RESAMPLER_SHAPES 

32 - RESAMPLER_DTYPES 

33 - RESAMPLER_RECORD_FIELDS 

34 

35 The DECISION constant should be a 

36 wepy.resampling.decisions.decision.Decision subclass. 

37 

38 This base class provides some hidden methods that are useful for 

39 various purposes. 

40 

41 To help maintain constraints on the number of walkers in a 

42 simulation the constructor allows for setting of a minimum and/or 

43 maximum of walkers. 

44 

45 These values are allowed to be either integers, None, or Ellipsis. 

46 

47 Integers set hard values for the minimum and maximum values. 

48 

49 None indicates that the min and max are unbounded. For the min 

50 this translates to a value of 1 since there must always be one 

51 walker. 

52 

53 Ellipsis is an indicator to determine the minimum and maximum 

54 dependent on the number of walkers provided for resampling. 

55 

56 If the max_num_walkers is Ellipsis and the number of walkers given 

57 is 10 then the maximum will be set to 10 for that 

58 resampling. Conversely, for if the minimum is Ellipsis. 

59 

60 If both the min and max are set to Ellipsis then the number of 

61 walkers is always kept the same. 

62 

63 Note that this does not implement any algorithm that actually 

64 decides how many walkers there will be but just checks that these 

65 constraints are met by those implementations in subclasses. 

66 

67 To allow for this checking the '_resample_init' method should be 

68 called at the beginning of the 'resample' method and the 

69 '_resample_cleanup' should be called at the end of the 'resample' 

70 method. 

71 

72 

73 

74 """

601 def resample(self, walkers, **kwargs): 

602 

603 self._resample_init(walkers=walkers) 

604 

605 n_walkers = len(walkers) 

606 

607 # the walker actions are all nothings with the same walker 

608 # index which is the default initialization 

609 resampling_data = self._init_walker_actions(n_walkers) 

610 

611 # normally decide is only for a single step and so does not 

612 # include the step_idx, so we add this to the records, and 

613 # convert the target idxs and decision_id to feature vector 

614 # arrays 

615 for walker_idx, walker_record in enumerate(resampling_data): 

616 walker_record['walker_idx'] = np.array([walker_idx]) 

617 walker_record['step_idx'] = np.array([0]) 

618 walker_record['walker_idx'] = np.array([walker_record['walker_idx']]) 

619 walker_record['decision_id'] = np.array([walker_record['decision_id']]) 

620 walker_record['target_idxs'] = np.array([walker_record['walker_idx']]) 

621 

622 # we only have one step so our resampling_records are just the 

623 # single list of walker actions 

624 resampling_data = resampling_data 

625 

626 # there is no change in state in the resampler so there are no 

627 # resampler records 

628 resampler_data = [{}] 

629 

630 # the resampled walkers are just the walkers 

631 

632 self._resample_cleanup(resampling_data=resampling_data, 

633 resampler_data=resampler_data, 

634 walkers=walkers) 

635 

636 return walkers, resampling_data, resampler_data
