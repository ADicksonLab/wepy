71 def _overlaps(self, positions, node_radii, node_idx): 

72 """ 

73 

74 Parameters 

75 ---------- 

76 positions : 

77 

78 node_radii : 

79 

80 node_idx : 

81 

82 

83 Returns 

84 ------- 

85 

86 """ 

87 

88 # get the nodes that this one overlaps with 

89 overlaps = [] 

90 for other_node_idx, other_node_position in enumerate(positions): 

91 

92 # you can't overlap yourself 

93 if node_idx == other_node_idx: 

94 overlaps.append(False) 

95 else: 

96 # check if there is an overlap between nodes 

97 diff = np.abs(positions[node_idx] - other_node_position) 

98 if diff < node_radii[node_idx] + node_radii[other_node_idx] + self.row_spacing: 

99 overlaps.append(True) 

100 else: 

101 overlaps.append(False) 

102 

103 return overlaps

105 def _node_row_length(self, node_positions, node_radii): 

106 """Get the edge to edge length of a row of nodes. 

107 

108 Parameters 

109 ---------- 

110 node_positions : 

111 

112 node_radii : 

113 

114 Returns 

115 ------- 

116 

117 row_length : float 

118 

119 """ 

120 

121 max_child_idx = np.argmax(node_positions) 

122 max_edge = node_positions[max_child_idx] + node_radii[max_child_idx] 

123 min_child_idx = np.argmin(node_positions) 

124 min_edge = node_positions[min_child_idx] - node_radii[min_child_idx] 

125 

126 return abs(max_edge - min_edge)

128 def _simple_gen_distribution(self, nodes_x, node_radii): 

129 """ 

130 

131 Parameters 

132 ---------- 

133 nodes_x : 

134 

135 node_radii : 

136 

137 

138 Returns 

139 ------- 

140 

141 new_nodes_positions 

142 

143 """ 

144 

145 # we want to update the positions given so we copy that array 

146 node_positions = np.array(copy(nodes_x)) 

147 

148 # first we check to see if there are any groups of nodes that 

149 # are in identical positions. If there are we can lump them 

150 # into a single node with a bigger radii and move things 

151 # around it 

152 

153 groups = [] 

154 group_positions = [] 

155 for node_idx, node_position in enumerate(node_positions): 

156 

157 # first get if this nodes position is already in the list 

158 # of known positions 

159 if node_position in group_positions: 

160 # if it is then get which group it is and add it to the list for that 

161 idx = group_positions.index(node_position) 

162 groups[idx].append(node_idx) 

163 

164 else: 

165 # start a new group for it 

166 group_positions.append(node_position) 

167 groups.append([node_idx]) 

168 

169 # now we filter out the groups that only have one node 

170 chosen_idxs = [idx for idx, group in enumerate(groups) 

171 if len(group) > 1] 

172 

173 group_positions = [group_positions[idx] for idx in chosen_idxs] 

174 

175 groups = [groups[idx] for idx in chosen_idxs] 

176 n_groups = len(groups) 

177 

178 # we make a collection of the nodes that are in groups 

179 grouped_node_idxs = set(it.chain(*groups)) 

180 

181 # then we remove these nodes from the positions and radii 

182 # arrays and replace them with single nodes of updated radii 

183 # and positions. 

184 group_radii = [] 

185 for group_idx, group in enumerate(groups): 

186 

187 # the new radii will be the sum of the diameters plus the 

188 # spacing between the nodes 

189 

190 # the sum of the diamters 

191 sum_diameters = sum([2 * node_radii[node_idx] 

192 for node_idx in group]) 

193 

194 # the spacing between is for 1 less than the total number 

195 # of the nodes in the group 

196 sum_spacing = self.row_spacing * (len(group) - 1) 

197 

198 # the whole groups radius 

199 group_radius = (sum_diameters + sum_spacing) / 2 

200 

201 group_radii.append(group_radius) 

202 

203 # then we rip out the nodes that are part of groups and are 

204 # left with the singletons 

205 singleton_idxs = [node_idx for node_idx, _ in enumerate(node_positions) 

206 if node_idx not in grouped_node_idxs] 

207 singleton_positions = [node_positions[node_idx] for node_idx in singleton_idxs] 

208 singleton_radii = [node_radii[node_idx] for node_idx in singleton_idxs] 

209 

210 # then we make the effective nodes to position later, we put 

211 # them at the beginning of the new positions list so we can 

212 # easily decompose it later and put them back out to how they 

213 # should be 

214 eff_positions = np.array(group_positions + singleton_positions) 

215 eff_radii = np.array(group_radii + singleton_radii) 

216 

217 # now that we have made the effective nodes we go ahead and 

218 # place them 

219 

220 # then for each node we place it sequentially 

221 for node_idx, node_radius in enumerate(eff_radii): 

222 

223 # get the overlaps for this node 

224 overlaps = self._overlaps(eff_positions, eff_radii, node_idx) 

225 

226 # get how many nodes overlap it to the left and to the 

227 # right of it's starting position 

228 left_overlaps = [i for i, _ in enumerate(overlaps) 

229 if eff_positions[i] < eff_positions[node_idx]] 

230 right_overlaps = [i for i, _ in enumerate(overlaps) 

231 if eff_positions[i] > eff_positions[node_idx]] 

232 

233 left_n_overlaps = len(left_overlaps) 

234 right_n_overlaps = len(right_overlaps) 

235 

236 # we just move all the other nodes to allow this node to 

237 # be placed where we proposed it 

238 

239 # left overlaps 

240 if left_n_overlaps > 0: 

241 

242 # repel the other nodes around it 

243 

244 # get the node idxs to the left of this one 

245 detangle_left_node_idxs = [i for i, position in enumerate(eff_positions) 

246 if position < eff_positions[node_idx]] 

247 

248 # get the (center to center) distances from this node to 

249 # all it's overlaps 

250 cc_left_overlap_dists = [abs(eff_positions[node_idx] - eff_positions[i]) 

251 for i in left_overlaps] 

252 

253 # then get the edge to edge distances (which can be 

254 # negative where they cross over) this is the sum of the 

255 # radii subtracted from the center to center distance (ee = d-(r_0 + r_1)) 

256 ee_left_overlap_dists = [cc_left_overlap_dists[i] - (node_radius + eff_radii[node_i]) 

257 for i, node_i in enumerate(left_overlaps)] 

258 

259 # we want the minimum one of the edge to edges (even if 

260 # its negative) 

261 left_min_ee_overlap_dist = min(ee_left_overlap_dists) 

262 

263 # this overlap should always be negative for the 

264 # overlapping nodes so we minimally need to move one 

265 # node this far away to have no overlap, on top of 

266 # that we add the spacing for the move distance to make 

267 left_move = -1 * (self.row_spacing + (-left_min_ee_overlap_dist)) 

268 

269 # then apply to those nodes 

270 eff_positions[detangle_left_node_idxs] += left_move 

271 

272 

273 # right overlaps 

274 if right_n_overlaps > 0: 

275 

276 # see above for comments 

277 detangle_right_node_idxs = [i for i, position in enumerate(eff_positions) 

278 if position > eff_positions[node_idx]] 

279 

280 cc_right_overlap_dists = [abs(eff_positions[node_idx] - eff_positions[i]) 

281 for i in right_overlaps] 

282 

283 ee_right_overlap_dists = [cc_right_overlap_dists[i] - (node_radius + eff_radii[node_i]) 

284 for i, node_i in enumerate(right_overlaps)] 

285 

286 right_min_ee_overlap_dist = min(ee_right_overlap_dists) 

287 

288 right_move = +1 * (self.row_spacing + (-right_min_ee_overlap_dist)) 

289 

290 eff_positions[detangle_right_node_idxs] += right_move 

291 

292 

293 # now that we have placed the effective nodes we want to 

294 # reconstruct the original nodes from the grouped ones 

295 

296 # make a new positions array that we can repopulate 

297 new_node_positions = [None for _ in node_positions] 

298 

299 # first for each group we need to place the nodes that had 

300 # identical positions within the space created by their larger 

301 # effective nodes 

302 for group_idx, group in enumerate(groups): 

303 

304 # the position of the group overall, since the grouped 

305 # nodes were at the front of the list of the effective 

306 # nodes with preserverd order we can use the group_index 

307 # to get them 

308 group_position = eff_positions[group_idx] 

309 group_radii = eff_radii[group_idx] 

310 

311 member_radii = [node_radii[node_idx] for node_idx in group] 

312 

313 # just place them like you would the initial distribution, 

314 # which is centered at 0 

315 member_positions = np.array(self._initial_parent_distribution(member_radii)) 

316 

317 # translate all nodes left (negative by the midpoint/radius) 

318 member_positions -= group_radii 

319 

320 # then we translate all of them to the overall group position 

321 member_positions += group_position 

322 

323 # now that we have good positions we can place them back 

324 # into the main level positions list 

325 for i, node_idx in enumerate(group): 

326 new_node_positions[node_idx] = member_positions[i] 

327 

328 # The singletons were at the end of the effective positions so 

329 # we have N_group_nodes + i where i is the index over the singletons 

330 for i, node_idx in enumerate(singleton_idxs): 

331 

332 new_node_positions[node_idx] = eff_positions[n_groups + i] 

333 

334 # sanity check that we covered them all 

335 assert all([True if pos is not None else False 

336 for pos in new_node_positions]),\ 

337 "not all positions recovered from the effective nodes" 

338 

339 

340 return new_node_positions

377 def _initial_parent_distribution(self, node_radii): 

378 """ 

379 

380 Parameters 

381 ---------- 

382 node_radii : 

383 

384 Returns 

385 ------- 

386 

387 positions 

388 

389 """ 

390 

391 # start at the origin and add nodes at positions that 

392 # accomodate their diameter and that are spaced by spacing_factor 

393 origin = 0.0 

394 

395 positions = [] 

396 

397 # place the first node at the origin, offset by its radius so 

398 # that the leading edge is at 0.0, then iterate for the rest 

399 # of them 

400 positions.append(origin + node_radii[0]) 

401 

402 # compute the edge so we know how much to space the next one 

403 # by 

404 last_edge = origin + 2 * node_radii[0] 

405 

406 # then do the rest 

407 for node_idx, node_radius in enumerate(node_radii[1:]): 

408 

409 node_idx += 1 

410 

411 # set the position from the last edge plus the space and 

412 # the new nodes radius 

413 position = last_edge + self.row_spacing + node_radius 

414 

415 positions.append(position) 

416 

417 # then calculate that ones edge 

418 last_edge = position + node_radius 

419 

420 return positions

422 def _center_row(self, positions, radii, center): 

423 """Centers the midpoint of a row of nodes around a number. 

424 

425 Parameters 

426 ---------- 

427 positions : 

428 

429 radii : 

430 

431 center : float 

432 

433 

434 Returns 

435 ------- 

436 

437 centered_positions 

438 

439 """ 

440 

441 # calculate the length of the row 

442 row_length = self._node_row_length(positions, radii) 

443 

444 min_child_idx = np.argmin(positions) 

445 min_edge = positions[min_child_idx] - radii[min_child_idx] 

446 

447 # row midpoint is the minimum edge plus the radius of the row 

448 row_midpoint = min_edge + (row_length / 2) 

449 

450 # translate all the points so that the midpoint is at 0 

451 centered_positions = positions - row_midpoint 

452 

453 return centered_positions

455 def _layout_array(self, parent_table, radii_array): 

456 """Generates the structured array of positions for nodes in the parent 

457 forest tree layout given the corresponding radii. 

458 

459 Parameters 

460 ---------- 

461 parent_table : 

462 

463 radii_array : 

464 

465 Returns 

466 ------- 

467 

468 node_positions 

469 

470 """ 

471 

472 n_timesteps = len(parent_table) 

473 n_walkers = len(parent_table[0]) 

474 

475 # initialize the positions to zeros, we add one to the 

476 # timesteps for the roots of the resampling trees 

477 node_positions = np.zeros((n_timesteps+1, n_walkers, 3)) 

478 

479 # initialize the first generation (cycle) node positions, this 

480 # is the root positions, we give it the radii of the first row in the array 

481 first_gen_positions = np.array(self._initial_parent_distribution(radii_array[0])) 

482 

483 # then center them around the central axis 

484 first_gen_positions = self._center_row(first_gen_positions, radii_array[0], 

485 self.central_axis) 

486 

487 # save them as full coordinates for visualization 

488 node_positions[0] = np.array([[x, 0.0, 0.0] for x in first_gen_positions]) 

489 

490 # we use the last gen positions to make the next gen 

491 last_gen_positions = first_gen_positions 

492 

493 # layout all the rest of the steps 

494 for step_idx in range(n_timesteps): 

495 

496 # the generation index is the one used for the actual tree 

497 # and it is always one more than the step index, since the 

498 # first generation is not counted as a step and was 

499 # already made outside the loop 

500 generation_idx = step_idx + 1 

501 

502 # generate the starting positions for the next generation nodes 

503 curr_gen_positions = self._simple_next_gen(last_gen_positions, 

504 parent_table[step_idx], 

505 radii_array[generation_idx]) 

506 

507 # center them around the central axis 

508 curr_gen_positions = self._center_row(curr_gen_positions, 

509 radii_array[generation_idx], 

510 self.central_axis) 

511 

512 

513 # figure out how big the increase in the Y direction 

514 # should be. This is based on the largest radii of the 

515 # last step and the largets radii of this step with the 

516 # mandatory spacing in between 

517 

518 # get the y dimension of the last step 

519 last_y = node_positions[generation_idx-1, 0, 1] 

520 

521 # get the largest radii of the last step 

522 last_max_radius = max(radii_array[generation_idx-1]) 

523 

524 # get the largest radii of this step 

525 this_max_radius = max(radii_array[generation_idx]) 

526 

527 # compute the step y dimension, which is from the last y 

528 # value plus the max radii plus the spacing 

529 step_y = last_y + last_max_radius + self.step_spacing + this_max_radius 

530 

531 # then generate the coordinates 

532 node_positions[generation_idx] = np.array( 

533 [np.array([x, step_y, 0.0]) 

534 for x in curr_gen_positions]) 

535 

536 # set the last gen positions 

537 last_gen_positions = curr_gen_positions 

538 

539 return node_positions

542 def layout(self, parent_forest, node_radii=None): 

543 """Given a parent forest object, returns a dictionary 

544 mapping nodes to their xyz layout coordinates. 

545 

546 If the node radii are given (as a dictionary mapping node ID 

547 to the desired radius) these are used as the the node radii 

548 and the default node radius parameter of the layout object is 

549 ignored. 

550 

551 Radii are needed to calculate the proper positions so there is 

552 no overlaps between nodes. 

553 

554 Parameters 

555 ---------- 

556 parent_forest : ParentForest object 

557 

558 node_radii : dict of node_id: float 

559 A dictionary mapping the nodes to node radii. 

560 (Default value = None) 

561 

562 

563 Returns 

564 ------- 

565 node_coords : dict of node_id: array_like of float of dim (3,) 

566 x, y, z coordinates for all nodes. Z will be 0 for all. 

567 

568 Raises 

569 ------ 

570 ValueError 

571 If an invalid node_id is given. 

572 

573 Warns 

574 ----- 

575 If some but not all nodes were given assigned radii. Uses 

576 default value for unspecified nodes. 

577 

578 """ 

579 

580 # get the parent table from the parent forest 

581 parent_table = parent_forest.parent_table 

582 

583 # default your node radii to the layout parameter, and fill in 

584 # the rest with the node_radii given to this method 

585 

586 # we need to generate an array of node radii, the size of 

587 # which is the parent table plus one extra row for the 

588 # root parents 

589 radii_array = [] 

590 

591 # count the number of nodes to check later whether or not the 

592 # user has given complete coverage with their radii. We can 

593 # issue a warning so that the user is not confused. 

594 n_nodes = 0 

595 

596 # the "root" row, it is the same length as the first in the 

597 # parent table 

598 radii_array.append([self.node_radius for col in parent_table[0]]) 

599 n_nodes += len(parent_table[0]) 

600 

601 # the rest of them 

602 for row in parent_table: 

603 radii_array.append([self.node_radius for col in row]) 

604 n_nodes += len(row) 

605 

606 # if any radii are given overwrite the default size, we count 

607 # how many new radii we give so that we can compare to the 

608 # actual number of nodes 

609 n_new_radii = 0 

610 if node_radii is not None: 

611 for node_id, radius in node_radii.items(): 

612 

613 # we know that the node_id is the (cycle_idx, 

614 # walker_idx), so we use these as indices on the array 

615 cycle_idx, walker_idx = node_id 

616 

617 # the indices in the node_array start at the -1 root 

618 # nodes so we increase cycle index by one to match, 

619 # this is called the generation index 

620 generation_idx = cycle_idx + 1 

621 

622 try: 

623 radii_array[generation_idx][walker_idx] = radius 

624 except IndexError: 

625 raise ValueError("invalid node_id was given") 

626 

627 n_new_radii += 1 

628 

629 # check if we have fewer assignments than there were nodes 

630 if n_new_radii < n_nodes: 

631 warn( 

632 "not all nodes were assigned custom radii," 

633 " default value {} was used instead".format(self.node_radius)) 

634 

635 layout_array = self._layout_array(parent_table, radii_array) 

636 

637 # make a dictionary mapping node ids to layout values 

638 node_coords = {} 

639 for cycle_idx, layout_row in enumerate(layout_array): 

640 

641 for walker_idx, coord in enumerate(layout_row): 

642 

643 # since the layout array starts at the root nodes (cycle 

644 # -1) we just reduce the cycle_idx by one when we set the 

645 # node attribute 

646 node_coords[(cycle_idx-1, walker_idx)] = coord 

647 

648 return node_coords
