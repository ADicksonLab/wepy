10def traj_box_vectors_to_lengths_angles(traj_box_vectors): 

11 """Convert box vectors for multiple 'frames' (a 'trajectory') to box lengths and angles. 

12 

13 Parameters 

14 ---------- 

15 traj_box_vectors : arraylike of float of shape (n_frames, n_dims, n_dims) 

16 Box vector matrices (x, y, z) for a trajectory of frames. 

17 

18 Returns 

19 ------- 

20 traj_box_lengths : arraylike of float of shape (n_frames, n_dims) 

21 The lengths of the box for each frame 

22 

23 traj_box_angles : arraylike of float of shape (n_frames, n_dims) 

24 The angles of the box vectors to normal for each frame in degrees. 

25 

26 """ 

27 

28 traj_unitcell_lengths = [] 

29 for basis in traj_box_vectors: 

30 traj_unitcell_lengths.append(np.array([np.linalg.norm(frame_v) for frame_v in basis])) 

31 

32 traj_unitcell_lengths = np.array(traj_unitcell_lengths) 

33 

34 traj_unitcell_angles = [] 

35 for vs in traj_box_vectors: 

36 

37 angles = np.array([np.degrees( 

38 np.arccos(np.dot(vs[i], vs[j])/ 

39 (np.linalg.norm(vs[i]) * np.linalg.norm(vs[j])))) 

40 for i, j in [(0,1), (1,2), (2,0)]]) 

41 

42 traj_unitcell_angles.append(angles) 

43 

44 traj_unitcell_angles = np.array(traj_unitcell_angles) 

45 

46 return traj_unitcell_lengths, traj_unitcell_angles

48def box_vectors_to_lengths_angles(box_vectors): 

49 """Convert box vectors for a single 'frame' to lengths and angles. 

50 

51 Parameters 

52 ---------- 

53 box_vectors : arraylike of float of shape (n_dims, n_dims) 

54 Box vector matrices (x, y, z) for a single frame. 

55 

56 Returns 

57 ------- 

58 box_lengths : arraylike of float of shape (n_dims,) 

59 The lengths of the box 

60 

61 box_angles : arraylike of float of shape (n_dims,) 

62 The angles of the box vectors to normal in degrees. 

63 

64 """ 

65 

66 # calculate the lengths of the vectors through taking the norm of 

67 # them 

68 unitcell_lengths = [] 

69 for basis in box_vectors: 

70 unitcell_lengths.append(np.linalg.norm(basis)) 

71 unitcell_lengths = np.array(unitcell_lengths) 

72 

73 # calculate the angles for the vectors 

74 unitcell_angles = np.array([np.degrees( 

75 np.arccos(np.dot(box_vectors[i], box_vectors[j])/ 

76 (np.linalg.norm(box_vectors[i]) * np.linalg.norm(box_vectors[j])))) 

77 for i, j in [(0,1), (1,2), (2,0)]]) 

78 

79 return unitcell_lengths, unitcell_angles

106def lengths_and_angles_to_box_vectors(a_length, b_length, c_length, alpha, beta, gamma): 

107 """Convert from the lengths/angles of the unit cell to the box 

108 vectors (Bravais vectors). The angles should be in degrees. 

109 

110 Parameters 

111 ---------- 

112 a_length : scalar or np.ndarray 

113 length of Bravais unit vector **a** 

114 b_length : scalar or np.ndarray 

115 length of Bravais unit vector **b** 

116 c_length : scalar or np.ndarray 

117 length of Bravais unit vector **c** 

118 alpha : scalar or np.ndarray 

119 angle between vectors **b** and **c**, in degrees. 

120 beta : scalar or np.ndarray 

121 angle between vectors **c** and **a**, in degrees. 

122 gamma : scalar or np.ndarray 

123 angle between vectors **a** and **b**, in degrees. 

124 

125 Returns 

126 ------- 

127 a : np.ndarray 

128 If the inputs are scalar, the vectors will one dimesninoal (length 3). 

129 If the inputs are one dimension, shape=(n_frames, ), then the output 

130 will be (n_frames, 3) 

131 b : np.ndarray 

132 If the inputs are scalar, the vectors will one dimesninoal (length 3). 

133 If the inputs are one dimension, shape=(n_frames, ), then the output 

134 will be (n_frames, 3) 

135 c : np.ndarray 

136 If the inputs are scalar, the vectors will one dimesninoal (length 3). 

137 If the inputs are one dimension, shape=(n_frames, ), then the output 

138 will be (n_frames, 3) 

139 

140 Examples 

141 -------- 

142 Notes 

143 ----- 

144 This code is adapted from gyroid, which is licensed under the BSD 

145 http://pythonhosted.org/gyroid/_modules/gyroid/unitcell.html 

146 >>> import numpy as np 

147 >>> result = lengths_and_angles_to_box_vectors(1, 1, 1, 90.0, 90.0, 90.0) 

148 """ 

149 if np.all(alpha < 2*np.pi) and np.all(beta < 2*np.pi) and np.all(gamma < 2*np.pi): 

150 warnings.warn('All your angles were less than 2*pi. Did you accidentally give me radians?') 

151 

152 alpha = alpha * np.pi / 180 

153 beta = beta * np.pi / 180 

154 gamma = gamma * np.pi / 180 

155 

156 a = np.array([a_length, np.zeros_like(a_length), np.zeros_like(a_length)]) 

157 b = np.array([b_length*np.cos(gamma), b_length*np.sin(gamma), np.zeros_like(b_length)]) 

158 cx = c_length*np.cos(beta) 

159 cy = c_length*(np.cos(alpha) - np.cos(beta)*np.cos(gamma)) / np.sin(gamma) 

160 cz = np.sqrt(c_length*c_length - cx*cx - cy*cy) 

161 c = np.array([cx,cy,cz]) 

162 

163 if not a.shape == b.shape == c.shape: 

164 raise TypeError('Shape is messed up.') 

165 

166 # Make sure that all vector components that are _almost_ 0 are set exactly 

167 # to 0 

168 tol = 1e-6 

169 a[np.logical_and(a>-tol, a<tol)] = 0.0 

170 b[np.logical_and(b>-tol, b<tol)] = 0.0 

171 c[np.logical_and(c>-tol, c<tol)] = 0.0 

172 

173 return a.T, b.T, c.T
