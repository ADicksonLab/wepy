64def mdtraj_to_json_topology(mdj_top): 

65 """Generate a JSON string topology from an mdtraj.Topology object. 

66 

67 This is the primary way in which JSON toplogies can be 

68 generated. 

69 

70 This format was created for use in the HDF5 format generated by 

71 the mdtraj library. 

72 

73 Parameters 

74 ---------- 

75 mdj_top : mdtraj.Topology object 

76 The topology you want to serialize to a JSON string. 

77 

78 Returns 

79 ------- 

80 json_top : str 

81 JSON string of the topology. 

82 

83 Warnings 

84 -------- 

85 The topology format for wepy is subject to change in the future! 

86 

87 Notes 

88 ----- 

89 Copied in part from MDTraj.formats.hdf5.topology setter. 

90 

91 """ 

92 

93 topology_dict = { 

94 'chains': [], 

95 'bonds': [] 

96 } 

97 

98 for chain in mdj_top.chains: 

99 chain_dict = { 

100 'residues': [], 

101 'index': int(chain.index) 

102 } 

103 for residue in chain.residues: 

104 residue_dict = { 

105 'index': int(residue.index), 

106 'name': str(residue.name), 

107 'atoms': [], 

108 "resSeq": int(residue.resSeq), 

109 'segmentID' : residue.segment_id 

110 } 

111 

112 for atom in residue.atoms: 

113 

114 try: 

115 element_symbol_string = str(atom.element.symbol) 

116 except AttributeError: 

117 element_symbol_string = "" 

118 

119 residue_dict['atoms'].append({ 

120 'index': int(atom.index), 

121 'name': str(atom.name), 

122 'element': element_symbol_string 

123 }) 

124 chain_dict['residues'].append(residue_dict) 

125 topology_dict['chains'].append(chain_dict) 

126 

127 for atom1, atom2 in mdj_top.bonds: 

128 topology_dict['bonds'].append([ 

129 int(atom1.index), 

130 int(atom2.index) 

131 ]) 

132 

133 top_json_str = json.dumps(topology_dict) 

134 

135 return top_json_str

137def json_to_mdtraj_topology(json_string): 

138 """Convert a JSON string topology to an mdtraj.Toplogy object. 

139 

140 This can be used to write out different topology formats from 

141 mdtraj. 

142 

143 Parameters 

144 ---------- 

145 json_string : str 

146 JSON string of the topology. 

147 

148 Returns 

149 ------- 

150 topology : mdtraj.Topology object 

151 

152 Warnings 

153 -------- 

154 The topology format for wepy is subject to change in the future! 

155 

156 Notes 

157 ----- 

158 Copied in part from MDTraj.formats.hdf5 topology property. 

159 

160 """ 

161 

162 topology_dict = json.loads(json_string) 

163 

164 topology = mdj.Topology() 

165 

166 for chain_dict in sorted(topology_dict['chains'], key=operator.itemgetter('index')): 

167 chain = topology.add_chain() 

168 for residue_dict in sorted(chain_dict['residues'], key=operator.itemgetter('index')): 

169 try: 

170 resSeq = residue_dict["resSeq"] 

171 except KeyError: 

172 resSeq = None 

173 warn('No resSeq information found in HDF file, defaulting to zero-based indices') 

174 try: 

175 segment_id = residue_dict["segmentID"] 

176 except KeyError: 

177 segment_id = "" 

178 residue = topology.add_residue(residue_dict['name'], chain, 

179 resSeq=resSeq, segment_id=segment_id) 

180 for atom_dict in sorted(residue_dict['atoms'], key=operator.itemgetter('index')): 

181 try: 

182 element = elem.get_by_symbol(atom_dict['element']) 

183 except KeyError: 

184 element = elem.virtual 

185 topology.add_atom(atom_dict['name'], element, residue) 

186 

187 atoms = list(topology.atoms) 

188 for index1, index2 in topology_dict['bonds']: 

189 topology.add_bond(atoms[index1], atoms[index2]) 

190 

191 return topology

194def _new_json_to_mdtraj_topology(json_string): 

195 """Another implementation of converting JSON to mdtraj. 

196 

197 TODO what is the difference between this and the other one? 

198 

199 Parameters 

200 ---------- 

201 json_string : str 

202 JSON string of the topology. 

203 

204 Returns 

205 ------- 

206 topology : mdtraj.Topology object 

207 

208 

209 """ 

210 

211 topology_dict = json.loads(json_string) 

212 

213 # start a new topology 

214 topology = mdj.Topology() 

215 

216 # to not destroy the indexing in this file we will first go 

217 # through the topology and get all the atom dictionaries (adding 

218 # attributes for the residue indices), then loop through them and 

219 # make residues and chains when needed. 

220 

221 atoms = {} 

222 residues = {} 

223 

224 # the chains will just be collections of residues 

225 chains = [] 

226 

227 # loop through all the chains 

228 for chain_dict in topology_dict['chains']: 

229 

230 # get the index and add the chain then save it 

231 chain_idx = chain_dict['index'] 

232 chain = topology.add_chain() 

233 chains[chain_idx] = chain 

234 

235 # then go through the residues here 

236 for residue_dict in chain_dict['residues']: 

237 

238 residue_idx = residue_dict['index'] 

239 

240 resSeq = residue_dict["resSeq"] 

241 segment_id = residue_dict["segmentID"] 

242 

243 

244 residue = topology.add_residue(residue_dict['name'], 

245 chain, 

246 resSeq=residue_dict['resSeq'], 

247 segment_id=residue_dict['segmentID']) 

248 

249 for atom_dict in sorted(residue_dict['atoms'], key=operator.itemgetter('index')): 

250 try: 

251 element = elem.get_by_symbol(atom_dict['element']) 

252 except KeyError: 

253 element = elem.virtual 

254 topology.add_atom(atom_dict['name'], element, residue) 

255 

256 for index1, index2 in topology_dict['bonds']: 

257 topology.add_bond(atoms[index1], atoms[index2]) 

258 

259 return topology

261def traj_fields_to_mdtraj(traj_fields, json_topology, rep_key='positions'): 

262 """Create an mdtraj.Trajectory object directly from a dictionary of 

263 fields values. 

264 

265 The time value in the trajectory will just be the frame indices 

266 for the positions. 

267 

268 Parameters 

269 ---------- 

270 

271 traj_fields : dict of str: values 

272 The values for the trajectory, must have a positions field 

273 specified by the `rep_key` kwarg, and a 'box_vectors' field. 

274 

275 json_topology : str 

276 JSON format topology of the positions. 

277 

278 rep_key : str 

279 The key that will be used to access the field that will be 

280 used as the positions in the trajectory, and must match the 

281 topology. 

282 

283 """ 

284 

285 topology = json_to_mdtraj_topology(json_topology) 

286 

287 req_fields = ['box_vectors', rep_key] 

288 assert [field in traj_fields for field in req_fields], \ 

289 "Fields must have the fields: {}".format(",".join(req_fields)) 

290 

291 unitcell_lengths, unitcell_angles = traj_box_vectors_to_lengths_angles( 

292 traj_fields['box_vectors']) 

293 

294 cycles = list(range(traj_fields['box_vectors'].shape[0])) 

295 traj = mdj.Trajectory(traj_fields[rep_key], topology, 

296 time=cycles, 

297 unitcell_lengths=unitcell_lengths, unitcell_angles=unitcell_angles) 

298 return traj
