107 def map(self, *args, **kwargs): 

108 """Map the 'segment_func' to args. 

109 

110 Parameters 

111 ---------- 

112 *args : list of list 

113 Each element is the argument to one call of 'segment_func'. 

114 

115 Returns 

116 ------- 

117 results : list 

118 The results of each call to 'segment_func' in the same order as input. 

119 

120 Examples 

121 -------- 

122 

123 >>> Mapper(segment_func=sum).map([(0,1,2), (3,4,5)]) 

124 [3, 12] 

125 

126 """ 

127 

128 # expand the generators for the args and kwargs 

129 args = [list(arg) for arg in args] 

130 kwargs = {key : list(kwarg) for key, kwarg in kwargs.items()} 

131 

132 segment_times = [] 

133 results = [] 

134 for arg_idx in range(len(args[0])): 

135 start = time.time() 

136 

137 # get just the args for this call to func 

138 call_args = [arg[arg_idx] for arg in args] 

139 call_kwargs = {key : value[arg_idx] for key, value in kwargs.items()} 

140 

141 # run the task, catch any errors and reraise as a 

142 # TaskException to satisfy the pattern 

143 try: 

144 

145 result = self._func(*call_args, **call_kwargs) 

146 

147 except Exception as task_exception: 

148 

149 # get the traceback for the exception 

150 tb = sys.exc_info()[2] 

151 

152 msg = "Exception '{}({})' caught in a task.".format( 

153 type(task_exception).__name__, task_exception) 

154 traceback_log_msg = \ 

155 """Traceback: 

156-------------------------------------------------------------------------------- 

157{} 

158-------------------------------------------------------------------------------- 

159 """.format(''.join(traceback.format_exception( 

160 type(task_exception), task_exception, tb)), 

161 ) 

162 

163 logging.critical(msg + '\n' + traceback_log_msg) 

164 

165 # raise a TaskException to distinguish it from the worker 

166 # errors with the metadata about the original exception 

167 

168 raise TaskException("Error occured during task execution, recovery not possible.", 

169 wrapped_exception=task_exception, 

170 tb=tb) 

171 

172 end = time.time() 

173 segment_time = end - start 

174 segment_times.append(segment_time) 

175 

176 results.append(result) 

177 

178 self._worker_segment_times[0] = segment_times 

179 

180 return results

455 super().init(num_workers=num_workers, segment_func=segment_func,

458 manager = self._mp_ctx.Manager() 

459 

460 # Establish communication queues 

461 

462 # A queue for errors 

463 self._exception_queue = manager.Queue() 

464 

465 # queue for the tasks we know the batch size so we don't need 

466 # a JoinableQueue 

467 self._task_queue = manager.Queue() 

468 

469 # results queue 

470 self._result_queue = manager.Queue() 

471 

472 # use pipes for communication channels between this parent 

473 # process and the children for sending specific interrupts 

474 # such as the signal to kill them. Note that the clean way to 

475 # end the process is to send poison pills on the task queue, 

476 # this is for other stuff. IRQ is a common abbreviation for 

477 # interrupts 

478 self._irq_parent_conns = [] 

479 

480 # Start workers, giving them all the queues 

481 self._workers = [] 

482 for i in range(self.num_workers): 

483 

484 # make a pipe to communicate with this worker for the int 

485 parent_conn, child_conn = self._mp_ctx.Pipe() 

486 self._irq_parent_conns.append(parent_conn) 

487 

488 # create the worker giving it all of the communication 

489 # channels 

490 worker = self.worker_type(i, 

491 self._task_queue, 

492 self._result_queue, 

493 self._exception_queue, 

494 child_conn, 

495 **self._worker_attributes) 

496 self._workers.append(worker) 

497 

498 # start the worker processes 

499 for worker in self._workers: 

500 worker.start() 

501 

502 logging.info("Worker process started as name: {}; PID: {}".format(worker.name, 

503 worker.pid)) 

504

506 # for SIGTERM signals that will clean up our children cleanly 

507 signal.signal(signal.SIGTERM, self._sigterm_shutdown)

510 

511 logging.critical("Received external SIGTERM, forcing shutdown.") 

512 

513 self.force_shutdown()

515 def force_shutdown(self, **kwargs): 

516 

517 logging.critical("Forcing shutdown") 

518 

519 # our primary job is to shut down all of the running processes 

520 # without just shutting down the queues and breaking the pipes 

521 

522 # to do this we send the kill signals to them on the kill 

523 # channel. 

524 

525 for worker_idx, worker in enumerate(self._workers): 

526 

527 logging.critical("Sending SIGTERM message on {} to worker {}".format( 

528 self._irq_parent_conns[worker_idx].fileno(), worker_idx)) 

529 

530 # send a kill message to the worker 

531 self._irq_parent_conns[worker_idx].send(signal.SIGTERM) 

532 

533 logging.critical("All kill messages sent to workers") 

534 

535 

536 

537 # check that all have exited 

538 alive_workers = [worker.is_alive() for worker in self._workers] 

539 worker_acks = {} 

540 worker_exitcodes = {} 

541 premature_exit = False 

542 while any(alive_workers) and not premature_exit: 

543 

544 for worker_idx, worker in enumerate(self._workers): 

545 

546 # ignore already known dead workers 

547 if not alive_workers[worker_idx]: 

548 continue 

549 

550 if worker.is_alive(): 

551 

552 # if it is still alive and we have an ack from it 

553 # just terminate. There is a bug in the code and 

554 # is out of our control 

555 if worker_idx in worker_acks: 

556 logging.debug("Ack received from {} but has not shut down".format(worker.name)) 

557 premature_exit = True 

558 

559 # otherwise we need to try and receive the ack 

560 elif self._irq_parent_conns[worker_idx].poll(1): 

561 

562 # receive the acknowledgement 

563 ack = self._irq_parent_conns[worker_idx].recv() 

564 

565 logging.debug("Received {} acknowledgement from {}".format(ack, 

566 worker.name)) 

567 

568 # make sure the ack is affirmative 

569 if ack is True: 

570 worker_acks[worker_idx] = ack 

571 

572 # if it is an exeption wrap it as a worker 

573 # error and use the os to kill the process 

574 elif issubclass(type(ack), Exception): 

575 

576 # wrap it as a worker exception 

577 exception = WorkerException(wrapped_exception=ack) 

578 worker_acks[worker_idx] = exception 

579 

580 logging.critical("{} not responding, terminating with SIGTERM".format( 

581 worker.name)) 

582 

583 worker.terminate() 

584 

585 

586 else: 

587 alive_workers[worker_idx] = False 

588 worker_exitcodes[worker_idx] = worker.exitcode 

589 

590 if any(alive_workers): 

591 logging.critical("Terminating main process with running workers {}".format( 

592 ','.join([str(worker_idx) for worker_idx in range(len(self._workers)) 

593 if alive_workers[worker_idx]])))

598 def cleanup(self, **kwargs): 

599 """Runtime post-simulation tasks. 

600 

601 This is run either at the end of a successful simulation or 

602 upon an error in the main process of the simulation manager 

603 call to `run_cycle`. 

604 

605 The Mapper class performs no actions here and all arguments 

606 are ignored. 

607 

608 """ 

609 

610 super().cleanup(**kwargs) 

611 

612 # send poison pills (Stop signals) to the queues to stop them in a nice way 

613 # and let them finish up 

614 for i in range(self.num_workers): 

615 self._task_queue.put((None, None)) 

616 

617 # delete the queues and workers 

618 self._task_queue = None 

619 self._result_queue = None 

620 self._workers = None

623 def map(self, *args, **kwargs): 

624 # docstring in superclass 

625 

626 map_process = self._mp_ctx.current_process() 

627 logging.info("Mapping from process {}; PID {}".format(map_process.name, map_process.pid)) 

628 

629 # make tuples for the arguments to each function call 

630 task_args = zip(*args) 

631 kwargs = {key : list(kwarg) for key, kwarg in kwargs.items()} 

632 

633 num_tasks = len(args[0]) 

634 # Enqueue the jobs 

635 for task_idx, task_arg in enumerate(task_args): 

636 

637 task_kwargs = {key : value[task_idx] for key, value in kwargs.items()} 

638 

639 # a task will be the actual task and its task idx so we can 

640 # sort them later 

641 self._task_queue.put((task_idx, self._make_task(*task_arg, **task_kwargs))) 

642 

643 

644 logging.info("Waiting for tasks to be run") 

645 

646 # poll the exception and result queues for results 

647 n_results_left = num_tasks 

648 results = [] 

649 while n_results_left > 0: 

650 

651 # first check if any errors came back but don't wait, 

652 # since the methods for querying whether it is empty or 

653 # not are not reliable we just try and if we don't get 

654 # anything we will come back around 

655 try: 

656 proc_name, pid, exception = self._exception_queue.get_nowait() 

657 except pyq.Empty: 

658 pass 

659 

660 else: 

661 

662 logging.error("Exception occured in process {}; pid {}.".format( 

663 proc_name, pid)) 

664 

665 # we can handle Task and Worker exceptions differently 

666 if type(exception) == TaskException: 

667 

668 logging.critical("Exception encountered in a task which is unrecoverable." 

669 "You will need to reconfigure your components in a stable manner.") 

670 

671 

672 self.force_shutdown() 

673 

674 logging.critical("Shutdown complete.") 

675 raise exception 

676 

677 elif type(exception) == WorkerException: 

678 

679 # we make just an error message to say that errors 

680 # in the worker may be due to the network or 

681 # something and could recover 

682 logging.error("Exception encountered in the work mapper worker process." 

683 "Recovery possible, see further messages.") 

684 

685 # However, the current implementation doesn't 

686 # support retries or whatever so we issue a 

687 # critical log informing that it has been elevated 

688 # to critical and will force shutdown 

689 logging.critical("Worker error mode resiliency not supported at this time." 

690 "Performing force shutdown and simulation ending.") 

691 

692 self.force_shutdown() 

693 

694 logging.critical("Shutdown complete.") 

695 raise exception 

696 

697 else: 

698 logging.critical("Unknown exception encountered.") 

699 

700 self.force_shutdown() 

701 

702 logging.critical("Shutdown complete.") 

703 

704 raise exception 

705 

706 

707 # attempt to get something off of the results queue 

708 try: 

709 result = self._result_queue.get_nowait() 

710 except pyq.Empty: 

711 pass 

712 

713 # if we get something handle it 

714 else: 

715 

716 logging.info("Retrieved result: {}".format(result)) 

717 results.append(result) 

718 

719 # reduce the counter so we know when we are done 

720 n_results_left -= 1 

721 

722 # sort the results according to their task_idx 

723 results.sort() 

724 

725 # save the task run times, so they can be accessed if desired, 

726 # after clearing the task times from the last mapping 

727 

728 # DEBUG: removing this because it should be set on init() 

729 #self._worker_segment_times = {i : [] for i in range(self.num_workers)} 

730 

731 for task_idx, worker_idx, task_time, result in results: 

732 self._worker_segment_times[worker_idx].append(task_time) 

733 

734 # then just return the values of the function 

735 return [result for task_idx, worker_idx, task_time, result in results]

811 def run(self): 

812 

813 logging.debug("{}: starting to run".format(self.name)) 

814 

815 # try to run the worker and it's task, except either class of 

816 # error that can come from it either from the worker 

817 # (WorkerException) or the task (TaskException) and communicate it 

818 # back to the main process 

819 

820 # if we get an exception there is some cleanup logic 

821 run_exception = None 

822 

823 try: 

824 

825 # run the worker, which will retrieve its task from the 

826 # queue attempt to run the task, and if it succeeds will 

827 # put the results on the result queue, if the task fails 

828 # it will catch it and wrap it as a task exception 

829 self._run_worker() 

830 

831 except TaskException as task_exception: 

832 

833 logging.error("{}: TaskException caught".format(self.name)) 

834 

835 run_exception = task_exception 

836 

837 # anything else is considered a WorkerException so take the 

838 # original exception and generate a worker exception from that 

839 except Exception as exception: 

840 

841 logging.debug("{}: WorkerError caught".format(self.name)) 

842 

843 

844 # get the traceback 

845 tb = sys.exc_info()[2] 

846 

847 msg = "Exception '{}({})' caught in a worker.".format( 

848 type(exception).__name__, exception) 

849 traceback_log_msg = \ 

850 """Traceback: 

851-------------------------------------------------------------------------------- 

852{} 

853-------------------------------------------------------------------------------- 

854 """.format(''.join(traceback.format_exception( 

855 type(exception), exception, tb)), 

856 ) 

857 

858 logging.error("{}:".format(self.name) + msg + '\n' + traceback_log_msg) 

859 

860 # raise a TaskError to distinguish it from the worker 

861 # errors with the metadata about the original exception 

862 

863 worker_exception = WorkerException( 

864 "Error occured during worker execution.", 

865 wrapped_exception=exception, 

866 tb=tb) 

867 

868 run_exception = worker_exception 

869 

870 # raise worker_exception 

871 if run_exception is not None:

873 logging.debug("{}: Putting exception on exception queue".format(self.name)) 

874 

875 # then put the exception and the traceback onto the queue 

876 # so we can communicate back to the parent process 

877 try: 

878 self._exception_queue.put((self.name, self.pid, run_exception))

916 def _run_worker(self): 

917 

918 # run the logic associated with communication and liveness of 

919 # the worker process itself, this is not necessarily a fatal 

920 # (critical) error and restarting a worker might resolve the 

921 # problem. This calls the _run_task method though which is 

922 # always critical since the logic in the code cannot be 

923 # disputed 

924 

925 # TODO remove when confirmed that this works 

926 # worker_process = mp.current_process() 

927 logging.info("{}: Worker process started as name: {}; PID: {}".format( 

928 self.name, self.name, self.pid)) 

929 

930 while True: 

931 

932 # check to see if there is any signals in the interrupt channel 

933 if self._irq_channel.poll(): 

934 # get the message 

935 message = self._irq_channel.recv() 

936 

937 logging.debug("{}: Received message from mapper on filehandle {}: {}".format( 

938 self.name, self._irq_channel.fileno(), message)) 

939 

940 # handle the message 

941 

942 # a None is a signal to kill the process 

943 # unconditionally 

944 if message is signal.SIGTERM: 

945 

946 self._shutdown() 

947 

948 # break from the event (while) loop and shut down 

949 break 

950 

951 # anything is not recognized and we will continue and 

952 # report back that we don't recognize the message with 

953 # a ValueError object 

954 else: 

955 logging.error("{}: Message not recognized, continuing operations and" 

956 " sending error to mapper".format(self.name)) 

957 self._irq_channel.send( 

958 ValueError( 

959 "Message: {} not recognized continuing operations".format( 

960 message))) 

961 

962 # get the next task 

963 try: 

964 task_idx, next_task = self._task_queue.get(block=False, timeout=None) 

965 

966 logging.debug("{}: Got task {}".format(self.name, task_idx)) 

967 

968 except pyq.Empty: 

969 

970 task_idx = None 

971 next_task = Ellipsis 

972 

973 

974 # # check for the poison pill which is the signal to stop 

975 if next_task is None: 

976 

977 logging.info('{}: received {} {}: FINISHED'.format( 

978 self.name, task_idx, next_task)) 

979 

980 # TODO remove since we aren't using joinble queue anymore 

981 # mark the poison pill task as done 

982 #self.task_queue.task_done() 

983 

984 # and exit the loop 

985 break 

986 

987 # only execute this if a task was actually receieved from 

988 # the queue; an Ellipsis indicates continue the loop 

989 elif next_task is not Ellipsis: 

990 

991 logging.info('{}; task_idx : {}; args : {} '.format( 

992 self.name, task_idx, next_task.args)) 

993 

994 # run the task 

995 start = time.time() 

996 

997 answer = self._run_task(next_task) 

998 

999 end = time.time() 

1000 task_time = end - start 

1001 

1002 logging.info('{}: task_idx : {}; COMPLETED in {} s'.format( 

1003 self.name, task_idx, task_time)) 

1004 

1005 # put the results into the results queue with it's task 

1006 # index so we can sort them later 

1007 self._result_queue.put((task_idx, self.worker_idx, task_time, answer))
