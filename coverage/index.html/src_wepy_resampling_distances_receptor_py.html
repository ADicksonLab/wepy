1"""Distance metrics related to common features of receptor-ligand 

2processes in molecular systems. 

3 

4The ReceptorDistance class is an abstract class that provides some 

5common functionality for normalizing reference states, providing 

6correct indices of receptor and ligand atoms, and a common image 

7function. 

8 

9Subclasses of ReceptorDistance need only implement the 

10'image_distance' function according to their needs. 

11 

12The UnbindingDistance is a useful metric for enhancing ligand movement 

13away from the reference bound state conformation. 

14 

15The RebindingDistance is a useful metric for enhancing the movement of 

16a ligand towards a reference state. 

17 

18""" 

19

33class ReceptorDistance(Distance): 

34 """Common abstract class for receptor-ligand molecular systems. 

35 

36 Any input walker state should have the attributes 'positions' and 

37 'box_vectors' and will first preprocess these states, by 

38 recentering the complex. Two component systems like 

39 receptor-ligands in periodic boxes can sometimes drift to 

40 different periodic images of the box making them unsuitable for 

41 RMSD calculations. 

42 

43 The 'image' method will align the receptor structures of the state 

44 to the reference state after the recentering. 

45 

46 """ 

47 

48 def __init__(self, ligand_idxs, binding_site_idxs, ref_state): 

49 """Construct a distance metric. 

50 

51 Parameters 

52 ---------- 

53 

54 ligand_idxs : arraylike of int 

55 The indices of the atoms from the 'positions' attribute of 

56 states that correspond to the ligand molecule. 

57 

58 binding_site_idxs : arraylike of int 

59 The indices of the atoms from the 'positions' attribute of 

60 states that correspond to the atoms of the binding site of 

61 the receptor. These are the atoms you want to perform 

62 alignment on and so can be a subset of a complete molecule. 

63 

64 ref_state : object implementing WalkerState 

65 The reference state all walker states will be aligned to 

66 with 'positions' (Nx3 dims) and 'box_vectors' (3x3 array) 

67 attributes. 

68 

69 """ 

70 

71 

72 # the idxs of the ligand and binding site from the whole state 

73 self._lig_idxs = ligand_idxs 

74 self._bs_idxs = binding_site_idxs 

75 

76 # number of atoms in each 

77 self._n_lig_atoms = len(self._lig_idxs) 

78 self._n_bs_atoms = len(self._bs_idxs) 

79 

80 # the idxs used for the whole image 

81 self._image_idxs = np.concatenate( (self._lig_idxs, self._bs_idxs) ) 

82 

83 # the idxs of the ligand and binding site within the image 

84 self._image_lig_idxs = np.arange(self._n_lig_atoms) 

85 self._image_bs_idxs = np.arange(self._n_lig_atoms, self._n_lig_atoms + self._n_bs_atoms) 

86 

87 # save the reference state's image so we can align all further 

88 # images to it 

89 

90 self.ref_image = self._unaligned_image(ref_state) 

91 

92 def _unaligned_image(self, state): 

93 """The preprocessing method of states. 

94 

95 First it groups the binding site and ligand into the same 

96 periodic box image and then centers the box around their 

97 mutual center of mass and returns only the positions of the 

98 binding site and ligand. 

99 

100 Parameters 

101 ---------- 

102 state : object implementing WalkerState 

103 State with 'positions' (Nx3 dims) and 'box_vectors' (3x3 

104 array) attributes. 

105 

106 Returns 

107 ------- 

108 

109 """ 

110 

111 # get the box lengths from the vectors 

112 box_lengths, box_angles = box_vectors_to_lengths_angles(state['box_vectors']) 

113 

114 # recenter the protein-ligand complex into the center of the 

115 # periodic boundary conditions 

116 

117 # regroup the ligand and protein in together 

118 grouped_positions = group_pair(state['positions'], box_lengths, 

119 self._bs_idxs, self._lig_idxs) 

120 

121 # then center them around the binding site 

122 centered_positions = center_around(grouped_positions, self._bs_idxs) 

123 

124 # slice these positions to get the image 

125 state_image = centered_positions[self._image_idxs] 

126 

127 return state_image 

128 

129 def image(self, state): 

130 """Transform a state to a receptor image. 

131 

132 A receptor image is one in which the binding site and ligand 

133 are first normalized to be within the same periodic box image 

134 and at the center of the box, and then only the binding site 

135 and ligand are retained. 

136 

137 Parameters 

138 ---------- 

139 state : object implementing WalkerState 

140 State with 'positions' (Nx3 dims) and 'box_vectors' (3x3 array) 

141 attributes. 

142 

143 Returns 

144 ------- 

145 

146 receptor_image : array of float 

147 The positions of binding site and ligand after 

148 preprocessing. 

149 

150 """ 

151 

152 # get the unaligned image 

153 state_image = self._unaligned_image(state) 

154 

155 # then superimpose it to the reference structure 

156 sup_image, _, _ = superimpose(self.ref_image, state_image, idxs=self._image_bs_idxs) 

157 

158 return sup_image

161class UnbindingDistance(ReceptorDistance): 

162 """Distance metric for measuring differences between walker states in 

163 regards to the RMSDs between ligands. 

164 

165 Images are produced using the ReceptorDistance.image method. The 

166 distance between images then is solely the RMSD of the ligand 

167 atoms. 

168 

169 Thus walkers were ligands are divergent in position will have 

170 higher distances. 

171 

172 """ 

173 

174 def image_distance(self, image_a, image_b): 

175 

176 # we calculate the rmsd of only the ligands between the 

177 # images 

178 lig_rmsd = calc_rmsd(image_a, image_b, idxs=self._image_lig_idxs) 

179 

180 return lig_rmsd

182class RebindingDistance(ReceptorDistance): 

183 """Distance metric for measuring differences between walker states in 

184 regards to the RMSDs between ligands. 

185 

186 Images are produced using the ReceptorDistance.image method. The 

187 distance between images then is the relative difference between 

188 the ligand RMSDs to the reference state. 

189 

190 """ 

191 

192 def image_distance(self, image_a, image_b): 

193 

194 # we calculate the rmsd of only the ligands between each image 

195 # and the reference 

196 state_a_rmsd = calc_rmsd(self.ref_image, image_a, idxs=self._image_lig_idxs) 

197 state_b_rmsd = calc_rmsd(self.ref_image, image_b, idxs=self._image_lig_idxs) 

198 

199 # then we get the absolute value of the reciprocals of these rmsd 

200 # values 

201 d = abs(1./state_a_rmsd - 1./state_b_rmsd) 

202 

203 return d
