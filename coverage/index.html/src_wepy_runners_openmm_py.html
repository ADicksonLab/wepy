1"""OpenMM molecular dynamics runner with accessory classes. 

2 

3OpenMM is a library with support for running molecular dynamics 

4simulations with specific support for fast GPU calculations. The 

5component based architecture of OpenMM makes it a perfect fit with 

6wepy. 

7 

8In addition to the principle OpenMMRunner class there are a few 

9classes here that make using OpenMM runner more efficient. 

10 

11First is a WalkerState class (OpenMMState) that wraps the openmm state 

12object directly, itself is a wrapper around the C++ 

13datastructures. This gives better performance by not performing copies 

14to a WalkerState dictionary. 

15 

16Second, is the OpenMMWalker which is identical to the Walker class 

17except that it enforces the state is an actual instantiation of 

18OpenMMState. Use of this is optional. 

19 

20Finally, is the OpenMMGPUWorker class. This is to be used as the 

21worker type for the WorkerMapper work mapper. This is necessary to 

22allow passing of the device index to OpenMM for which GPU device to 

23use. 

24 

25"""

153 Regarding the 'enforce_box' option. 

154 

155 When retrieving states from an OpenMM simulation Context, you 

156 have the option to enforce periodic boundary conditions in the 

157 resulting atomic positions in a topology aware way that 

158 doesn't break bonds through boundaries. This is convenient for 

159 post-processing as this can be a complex task and is not

161

163 are ignored (such as implicit solvent ones) despite there 

164 being no option to not have periodic boundaries in the context 

165 itself. Likely if you are running one of these kinds of 

166 simulations you will not pay attention to the box vectors at 

167 all and the random defaults that exist will be very wrong but 

168 this incorrectness will not show in a non-wepy simulation with 

169 openmm unless you are handling the context states 

170 yourself. Then when you run in wepy the default of True to 

171 enforce the boxes will be applied and confusingly wrong 

172 answers will result that are difficult to find root cause of.

494class OpenMMState(WalkerState): 

495 """Walker state that wraps an simtk.openmm.State object. 

496 

497 The keys for which values in the state are available are given by 

498 the KEYS module constant (accessible through the class constant of 

499 the same name as well). 

500 

501 Additional fields can be added to these states through passing 

502 extra kwargs to the constructor. These will be automatically given 

503 a suffix of "_OTHER" to avoid name clashes. 

504 

505 """ 

506 

507 KEYS = KEYS 

508 """The provided attribute keys for the state.""" 

509 

510 OTHER_KEY_TEMPLATE = "{}_OTHER" 

511 """String formatting template for attributes not set in KEYS."""

513 def __init__(self, sim_state, **kwargs): 

514 """Constructor for OpenMMState. 

515 

516 Parameters 

517 ---------- 

518 state : simtk.openmm.State object 

519 The simulation state retrieved from the simulation constant. 

520 

521 kwargs : optional 

522 

523 Additional attributes to set for the state. Will add the 

524 "_OTHER" suffix to the keys 

525 

526 """ 

527 

528 # save the simulation state 

529 self._sim_state = sim_state 

530 

531 # save additional data if given 

532 self._data = {} 

533 for key, value in kwargs.items(): 

534 

535 # if the key is already in the sim_state keys we need to 

536 # modify it and raise a warning 

537 if key in self.KEYS: 

538 

539 warn("Key {} in kwargs is already taken by this class, renaming to {}".format( 

540 self.OTHER_KEY_TEMPLATE).format(key)) 

541 

542 # make a new key 

543 new_key = self.OTHER_KEY_TEMPLATE.format(key) 

544 

545 # set it in the data 

546 self._data[new_key] = value 

547 

548 # otherwise just set it 

549 else: 

550 self._data[key] = value

558 

559 # if this was a key for data not mapped from the OpenMM.State

563 (not key.startswith('parameter_derivatives'))

566 

567 # otherwise we have to specifically get the correct data and

569 else:

573 elif key == 'velocities':

577 elif key == 'kinetic_energy':

579 elif key == 'potential_energy':

587

590 parameters_dict = self.parameters_values() 

591 if parameters_dict is None: 

592 return None 

593 else:

946 # a proper state where all __getitem__ things are arrays 

947 def _dict_attr_to_compound_key_dict(self, root_key, attr_dict): 

948 """Transform a dictionary of values within the compound key 'root_key' 

949 to a dictionary mapping compound keys to values. 

950 

951 For example give the root_key 'parameters' and the parameters 

952 dictionary {'paramA' : 1.234} returns {'parameters/paramA' : 1.234}. 

953 

954 Parameters 

955 ---------- 

956 root_key : str 

957 The compound key prefix 

958 attr_dict : dict of str : value 

959 The dictionary with simple keys within the root key namespace. 

960 

961 Returns 

962 ------- 

963 compound_key_dict : dict of str : value 

964 The dictionary with the compound keys. 

965 

966 """ 

967 

968 key_template = "{}/{}" 

969 cmpd_key_d = {} 

970 for key, value in attr_dict.items(): 

971 

972 new_key = key_template.format(root_key, key) 

973 # if this is a proper feature 

974 if type(value) == np.ndarray: 

975 cmpd_key_d[new_key] = value 

976 elif hasattr(value, '__getitem__'): 

977 cmpd_key_d.update(self._dict_attr_to_compound_key_dict(new_key, value)) 

978 else: 

979 raise TypeError("Unsupported attribute type") 

980 

981 return cmpd_key_d

983 def _get_nested_attr_from_compound_key(self, compound_key, compound_feat_dict): 

984 """Get arbitrarily deeply nested compound keys from the full 

985 dictionary tree. 

986 

987 Parameters 

988 ---------- 

989 compound_key : str 

990 Compound key separated by '/' characters 

991 

992 compound_feat_dict : dict 

993 Dictionary of arbitrary depth 

994 

995 Returns 

996 ------- 

997 value 

998 Value requested by the key. 

999 

1000 """ 

1001 

1002 key_components = compound_key.split('/') 

1003 

1004 # if there is only one component of the key then it is not 

1005 # really compound, we won't complain just return the 

1006 # "dictionary" if it is not actually a dict like 

1007 if not hasattr(compound_feat_dict, '__getitem__'): 

1008 raise TypeError("Must provide a dict-like with the compound key") 

1009 

1010 value = compound_feat_dict[key_components[0]] 

1011 

1012 # if the value itself is compound recursively fetch the value 

1013 if hasattr(value, '__getitem__') and len(key_components[1:]) > 0: 

1014 

1015 subgroup_key = '/'.join(key_components[1:]) 

1016 

1017 return self._get_nested_attr_from_compound_key(subgroup_key, value) 

1018 

1019 elif hasattr(value, '__getitem__') and len(key_components[1:]) < 1: 

1020 raise ValueError("Key does not reference a leaf node of attribute") 

1021 

1022 # otherwise we have the right key so return the object 

1023 else: 

1024 return value
