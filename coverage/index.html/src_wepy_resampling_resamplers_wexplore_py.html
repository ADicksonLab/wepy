23def calc_squashable_walkers_single_method(walker_weights, max_weight): 

24 """Calculate the maximum number of squashable walkers in collection of 

25 walkers, that still satisfies the max weight constraint. 

26 

27 We don't guarantee or know if this is a completely optimal solver 

28 for this problem but it is assumed to be good enough in practice 

29 and no harm comes from underestimating it only a reduced potential 

30 performance. 

31 

32 Parameters 

33 ---------- 

34 walker_weights : list of float 

35 The weights of the walkers 

36 

37 max_weight : float 

38 The maximum weight a walker can have. 

39 

40 Returns 

41 ------- 

42 

43 n_squashable : int 

44 The maximum number of squashable walkers. 

45 

46 """ 

47 

48 

49 # to get an estimate of the number of squashable walkers we start 

50 # summing the weights starting from the smallest walker. When the 

51 # addition of the next highest weight walker would make the total 

52 # greater than max_weight then we quit and say that the number of 

53 # squashable walkers is the number of them summed up, minus one 

54 # for the fact that one of them won't be squashed if a merge of 

55 # all of them was to occur 

56 n_squashable = 0 

57 

58 # there must be at least 2 walkers in order to be able to do a 

59 # merge, so if there are not enough the number of squashable 

60 # walkers is 0 

61 if len(walker_weights) < 2: 

62 return n_squashable 

63 

64 

65 # sort the weights smallest to biggest 

66 walker_weights.sort() 

67 

68 idx = 0 

69 sum_weights = walker_weights[idx] 

70 merge_size = 1 

71 while sum_weights <= max_weight: 

72 

73 # if the next index would be out of bounds break out of the 

74 # loop 

75 if idx + 1 >= len(walker_weights): 

76 break 

77 else: 

78 idx += 1 

79 

80 # add this walker to the sum weights 

81 sum_weights += walker_weights[idx] 

82 

83 # add one to the merge size (since we only will make our 

84 # estimate based on the single largest possible merge) 

85 merge_size += 1 

86 

87 

88 else: 

89 # the loop condition failed so we remove the last count of 

90 # merge size from the merge group. This won't run if we break 

91 # out of the loop because of we are out of walkers to include 

92 merge_size -= 1 

93 

94 

95 # then we also take one less than that as the number of 

96 # squashable walkers 

97 n_squashable = merge_size - 1 

98 

99 

100 return n_squashable

104def decide_merge_groups_single_method(walker_weights, balance, max_weight): 

105 """Use the 'single' method for determining the merge groups. 

106 

107 Determine a solution to the backpack-like problem of assigning 

108 squashed walkers to KEEP_MERGE walkers. 

109 

110 The single method just assigns all squashed walkers in the 

111 collection to a single walker, thus there is a single merge_group. 

112 

113 Parameters 

114 ---------- 

115 walker_weights : list of float 

116 The weights of the walkers. 

117 

118 balance : int 

119 The net change in the number of walkers we desire. 

120 

121 max_weight : float 

122 The maximum weight a single walker can be. 

123 

124 Returns 

125 ------- 

126 

127 merge_groups : list of list of int 

128 The merge group solution. 

129 

130 result : bool 

131 Whether the merge group exceeds the max weight. 

132 

133 """ 

134 

135 assert balance < 0, "target balance must be negative" 

136 

137 # the number of walkers we need to choose in order to be 

138 # able to do the required amount of merges 

139 num_merge_walkers = abs(balance) + 1 

140 

141 # select the lowest weight walkers to use for merging, these 

142 # are idxs on the mergeable walkers and not the walker_idxs 

143 chosen_idxs = np.argsort(walker_weights)[:num_merge_walkers] 

144 

145 # check that this is not greater than the max weight 

146 if sum([walker_weights[chosen_idx] for chosen_idx in chosen_idxs]) > max_weight: 

147 result = False 

148 else: 

149 result = True 

150 

151 # return the chosen idxs as the sole full merge group 

152 return [chosen_idxs], result

721 """ 

722 

723 Parameters 

724 ---------- 

725 pmax : 

726  

727 root : 

728  

729 weights : 

730  

731

733 ------- 

734

738 walker_idxs = [i for i, weight in enumerate(weights)]

829 def _propagate_and_balance_shares(self, parental_balance, children_node_ids): 

830 """ 

831 

832 Parameters 

833 ---------- 

834 parental_balance : 

835  

836 children_node_ids : 

837  

838 

839 Returns 

840 ------- 

841 

842 """ 

843 

844 

845 # talk about "shares" which basically are the number of 

846 # slots/replicas that will be allocated to this region for 

847 # running sampling on 

848 

849 # we get the current number of shares for each child 

850 orig_children_shares = {child_id : len(self.node[child_id]['walker_idxs']) 

851 for child_id in children_node_ids} 

852

854 children_shares = copy(orig_children_shares) 

855

861 children_receivable_shares = {child_id : self.node[child_id]['n_possible_clones'] 

862 for child_id in children_node_ids} 

863

865 # balance to it's children in a simply valid manner

868 children_donatable_shares, 

869 children_receivable_shares)

890 children_donatable_shares,

893

900

902 "The balances of the child nodes ({}) do not balance to the parental balance ({})".format(

913 def _dispense_parental_shares(self, parental_balance, children_shares, 

914 children_donatable_shares, 

915 children_receivable_shares): 

916 """Given a parental balance and a set of children nodes, we dispense 

917 the shares indicated by the balance to the children nodes in a 

918 VALID but not necessarily optimal or desirable way. This 

919 merely checks for the hard constraints on the number of shares 

920 a region can either give or receive based on their capacity to 

921 clone and merge walkers. 

922  

923 An additional balancing step can be performed to redistribute them. 

924 

925 Parameters 

926 ---------- 

927 parental_balance : 

928  

929 children_shares : 

930  

931 children_donatable_shares : 

932  

933 children_receivable_shares : 

934  

935 

936 Returns 

937 ------- 

938 

939 """ 

940 

941 # this will be the totaled up dispensations for each child 

942 # region 

943 children_dispensations = {child_id : 0 for child_id in children_shares.keys()} 

944 

945 # if there is only one child it just inherits all of the 

946 # balance no matter what 

947 if len(children_shares.keys()) == 1: 

948 

949 child_node_id = list(children_shares.keys())[0] 

950 

951 # we put the shares for this only child in a dictionary 

952 # like the other methods would produce 

953 children_dispensations[child_node_id] = parental_balance 

954 

955 # there are more than one child so we accredit balances 

956 # between them 

957 elif len(children_shares.keys()) > 1: 

958 

959 # if the parent has a non-zero balance we either 

960 # increase (clone) or decrease (merge) the balance 

961 

962 # these poor children are inheriting a debt and must 

963 # decrease the total number of their shares :( 

964 if parental_balance < 0: 

965 

966 children_dispensations = self._dispense_debit_shares(parental_balance, 

967 children_shares, 

968 children_donatable_shares) 

969 

970 # these lucky children are inheriting a positive number of 

971 # shares!! :) 

972 elif parental_balance > 0: 

973 

974 children_dispensations = self._dispense_credit_shares(parental_balance, 

975 children_shares, 

976 children_receivable_shares) 

977 

978 else: 

979 raise RegionTreeError("no children nodes to give parental balance") 

980 

981 return children_dispensations

983 def _dispense_debit_shares(self, parental_balance, children_shares, 

984 children_donatable_shares): 

985 """For a negative parental balance we dispense it to the children 

986 nodes 

987 

988 Parameters 

989 ---------- 

990 parental_balance : 

991  

992 children_shares : 

993  

994 children_donatable_shares : 

995  

996 

997 Returns 

998 ------- 

999 

1000 """ 

1001 

1002 children_donatable_shares = copy(children_donatable_shares) 

1003 

1004 children_dispensations = {child_id : 0 for child_id in children_shares.keys()} 

1005 

1006 # list of the keys so we can iterate through them 

1007 children_node_ids = list(children_shares.keys()) 

1008 

1009 # dispense the negative shares as quickly as possible, 

1010 # they will be balanced later 

1011 child_iter = iter(children_node_ids) 

1012 remaining_balance = parental_balance 

1013 while remaining_balance < 0: 

1014 # get the node id 

1015 try: 

1016 child_node_id = next(child_iter) 

1017 except StopIteration: 

1018 # if the parental balance is still not zero and there 

1019 # are no more children then the children cannot 

1020 # balance it given their constraints and there is an 

1021 # error 

1022 raise RegionTreeError("Children cannot pay their parent's debt") 

1023 

1024 n_donatable = children_donatable_shares[child_node_id] 

1025 

1026 # if this child has any squashable walkers 

1027 if n_donatable > 0: 

1028 

1029 # we use those for paying the parent's debt 

1030 

1031 # the amount of the parental debt that can be 

1032 # paid (the payment) for this child region is 

1033 # either the number of squashable walkers or 

1034 # the absolute value of the parental balance 

1035 # (since it is negative for debts), whichever 

1036 # is smaller 

1037 payment = min(n_donatable, abs(remaining_balance)) 

1038 

1039 # take this from the remaining balance 

1040 remaining_balance += payment 

1041 

1042 # and take it away from the childs due balance and shares 

1043 children_dispensations[child_node_id] -= payment 

1044 

1045 # also account for that in its donatable shares 

1046 children_donatable_shares[child_node_id] -= payment 

1047 

1048 # double check the balance is precisely 0, we want to 

1049 # dispense all the shares as well as not accidentally 

1050 # overdispensing 

1051 assert remaining_balance == 0, "balance is not 0" 

1052 

1053 

1054 return children_dispensations

1056 def _dispense_credit_shares(self, parental_balance, children_shares, 

1057 children_receivable_shares): 

1058 """ 

1059 

1060 Parameters 

1061 ---------- 

1062 parental_balance : 

1063  

1064 children_shares : 

1065  

1066 children_receivable_shares : 

1067  

1068 

1069 Returns 

1070 ------- 

1071 

1072 """ 

1073 

1074 children_receivable_shares = copy(children_receivable_shares) 

1075 

1076 children_dispensations = {child_id : 0 for child_id in children_shares.keys()} 

1077 

1078 # list of the keys so we can iterate through them 

1079 children_node_ids = list(children_shares.keys()) 

1080 

1081 # dispense the shares to the able children as quickly 

1082 # as possible, they will be redistributed in the next 

1083 # step 

1084 child_iter = iter(children_node_ids) 

1085 remaining_balance = parental_balance 

1086 while remaining_balance > 0: 

1087 

1088 # get the node id 

1089 try: 

1090 child_node_id = next(child_iter) 

1091 except StopIteration: 

1092 # if the parental balance is still not zero and there 

1093 # are no more children then the children cannot 

1094 # balance it given their constraints and there is an 

1095 # error 

1096 raise RegionTreeError("Children cannot accept their parent's credit") 

1097 

1098 # give as much of the parental balance as we can 

1099 # to the walkers. In the next step all this 

1100 # balance will be shared among the children so all 

1101 # we need to do is dispense all the shares without 

1102 # care as to who gets them, as long as they can 

1103 # keep it 

1104 n_receivable = children_receivable_shares[child_node_id] 

1105 

1106 # the amount to be disbursed to this region is 

1107 # either the number of possible clones (the 

1108 # maximum it can receive) or the full parental 

1109 # balance, whichever is smaller 

1110 disbursement = min(n_receivable, abs(remaining_balance)) 

1111 

1112 # give this disbursement by taking away from the 

1113 # positive balance 

1114 remaining_balance -= disbursement 

1115 

1116 # add these shares to the net balances and share 

1117 # totals 

1118 children_dispensations[child_node_id] += disbursement 

1119 

1120 # also subtract this from the number of receivable shares 

1121 # for the child 

1122 children_receivable_shares[child_node_id] -= disbursement 

1123 

1124 # double check the balance is precisely 0, we want to 

1125 # dispense all the shares as well as not accidentally 

1126 # overdispensing 

1127 assert remaining_balance == 0, "balance is not 0" 

1128 

1129 return children_dispensations

1131 def _balance_children_shares(self, children_shares, 

1132 children_donatable_shares, 

1133 children_receivable_shares): 

1134 """Given a dictionary mapping the child node_ids to the total number 

1135 of shares they currently hold we balance between them in order 

1136 to get an even distribution of the shares as possible. 

1137 

1138 Parameters 

1139 ---------- 

1140 children_shares : 

1141  

1142 children_donatable_shares : 

1143  

1144 children_receivable_shares : 

1145  

1146 

1147 Returns 

1148 ------- 

1149 

1150 """ 

1151 

1152 children_shares = copy(children_shares) 

1153 

1154 # generate the actual donation pair and the amount that should 

1155 # be donated for the best outcome 

1156 donor_node_id, acceptor_node_id, donation_amount = \ 

1157 self._gen_best_donation(children_shares, 

1158 children_donatable_shares, 

1159 children_receivable_shares) 

1160 

1161 # if the donation amount is zero we make no donation 

1162 if donation_amount > 0: 

1163 

1164 # account for this donation in the shares 

1165 children_shares[donor_node_id] -= donation_amount 

1166 children_shares[acceptor_node_id] += donation_amount 

1167 

1168 # subtract the donation donatable_shares from the donor 

1169 # and add the donation to the donatable_shares of the 

1170 # acceptor 

1171 children_donatable_shares[donor_node_id] -= donation_amount 

1172 children_donatable_shares[acceptor_node_id] += donation_amount 

1173 

1174 # do the opposite to the receivable shares 

1175 children_receivable_shares[donor_node_id] += donation_amount 

1176 children_receivable_shares[acceptor_node_id] -= donation_amount 

1177 

1178 # we have decided the first donation, however more will be 

1179 # performed as long as the amount of the donation is either 0 

1180 # or that two donations of only 1 share occur twice in a 

1181 # row. The former occurs in scenarios when there is an even 

1182 # balance and the latter in an odd scenario and the last odd 

1183 # share would get passed back and forth 

1184 

1185 # we keep track of the previous donation, and initialize it to 

1186 # None for now 

1187 previous_donation_amount = donation_amount 

1188 

1189 while (donation_amount > 0) and \ 

1190 not (previous_donation_amount == 1 and donation_amount == 1): 

1191 

1192 # update the previous donation amount 

1193 previous_donation_amount = donation_amount 

1194 

1195 # get the next best donation 

1196 donor_node_id, acceptor_node_id, donation_amount = \ 

1197 self._gen_best_donation(children_shares, 

1198 children_donatable_shares, 

1199 children_receivable_shares) 

1200 

1201 # if there is a donation to be made make it 

1202 if donation_amount > 0: 

1203 

1204 # account for this donation in the shares 

1205 children_shares[donor_node_id] -= donation_amount 

1206 children_shares[acceptor_node_id] += donation_amount 

1207 

1208 # subtract the donation donatable_shares from the donor 

1209 # and add the donation to the donatable_shares of the 

1210 # acceptor 

1211 children_donatable_shares[donor_node_id] -= donation_amount 

1212 children_donatable_shares[acceptor_node_id] += donation_amount 

1213 

1214 # do the opposite to the receivable shares 

1215 children_receivable_shares[donor_node_id] += donation_amount 

1216 children_receivable_shares[acceptor_node_id] -= donation_amount 

1217 

1218 

1219 

1220 return children_shares

1223 def _gen_best_donation(self, children_shares, 

1224 children_donatable_shares, 

1225 children_receivable_shares): 

1226 """Given a the children shares generate the best donation. Returns the 

1227 donor_node_id the acceptor_node_id and the donation that 

1228 should be done between them and that will be guaranteed to be 

1229 valid. (this is done by checking the attributes of the regions 

1230 node however, no changes to node state are performed) 

1231  

1232 returns donor_node_id, acceptor_node_id, donation_amount 

1233 

1234 Parameters 

1235 ---------- 

1236 children_shares : 

1237  

1238 children_donatable_shares : 

1239  

1240 children_receivable_shares : 

1241  

1242 

1243 Returns 

1244 ------- 

1245 

1246 """ 

1247 

1248 # to find the best possible donation we would like to give 

1249 # shares from the region with the most to the region with the 

1250 # least and give as many as possible that will equalize them, 

1251 # however the size of a donation is dependent not only on the 

1252 # amount of shares each region has but also the number of 

1253 # squashable walkers and the number of possible clones that 

1254 # satisfy the maximum and minimum walker weight 

1255 # constraints. These are given by the 

1256 # children_donatable_shares and children_receivable_shares. We 

1257 # use arguments instead of accessing the attributes of the 

1258 # object because so this can be done in an iterative manner 

1259 # before modifying the node attributes. 

1260 

1261 # default values for the results 

1262 best_pair = (None, None) 

1263 best_donation_amount = 0 

1264 

1265 # if there are not enough children regions to acutally make 

1266 # pairings between then we just return the default negative 

1267 # result 

1268 if len(children_shares) < 2: 

1269 donor_node_id, acceptor_node_id = best_pair 

1270 return donor_node_id, acceptor_node_id, best_donation_amount 

1271 

1272 

1273 # we want to get the list of pairings where each pairing is 

1274 # (donor, acceptor) 

1275 pairings = [] 

1276 for a, b in it.combinations(children_shares.keys(), 2): 

1277 

1278 

1279 # find the largest difference comparing (a,b) and (b,a), 

1280 # this will give the donor, acceptor pair 

1281 permutations = [(a,b), (b,a)] 

1282 perm_idx = np.argmax([children_shares[i] - children_shares[j] 

1283 for i, j in permutations]) 

1284 

1285 donor_acceptor_pair = permutations[perm_idx] 

1286 

1287 pairings.append(donor_acceptor_pair) 

1288 

1289 # to find the best match we first calculate the differences in 

1290 # the number of shares for each possible pairing between 

1291 # children shares 

1292 pairings_differences = [children_shares[donor_id] - children_shares[acceptor_id] 

1293 for donor_id, acceptor_id in pairings] 

1294 

1295 pairings_donations = [] 

1296 # then we find all the non-zero pairings 

1297 for i, difference in enumerate(pairings_differences): 

1298 

1299 # if there is a positive difference then we calculate what 

1300 # the largest donation would be 

1301 if difference > 0: 

1302 

1303 donor_node_id, acceptor_node_id = pairings[i] 

1304 

1305 # get the total numbers of shares for each 

1306 donor_n_shares = children_shares[donor_node_id] 

1307 acceptor_n_shares = children_shares[acceptor_node_id] 

1308 

1309 # as well as the donatable and receivable shares 

1310 donor_donatable_shares = children_donatable_shares[donor_node_id] 

1311 acceptor_receivable_shares = children_receivable_shares[acceptor_node_id] 

1312 

1313 # actually calculate the maximum donation 

1314 donation_amount = self._calc_share_donation(donor_n_shares, acceptor_n_shares, 

1315 donor_donatable_shares, 

1316 acceptor_receivable_shares) 

1317 

1318 pairings_donations.append(donation_amount) 

1319 

1320 # if there is no difference then the donation amount will also be zero 

1321 else: 

1322 pairings_donations.append(0) 

1323 

1324 # now we can zip them all together and then sort them such 

1325 # that we first sort on the size of the number of shares and 

1326 # then on the size of the donation 

1327 pair_values = list(zip(pairings_differences, pairings_donations, pairings)) 

1328 pair_values.sort() 

1329 # largest to smallest 

1330 pair_values.reverse() 

1331 

1332 # now we take the pairing that has the highest difference and 

1333 # has a nonzero donation size. Note there may be other 

1334 # pairings with the same numbers that will just be ignored. 

1335 pair_iter = iter(pair_values) 

1336 

1337 

1338 # loop through until the first best donation is found. 

1339 # get the first pairing 

1340 try: 

1341 diff, donation, pairing = next(pair_iter) 

1342 except StopIteration: 

1343 raise RegionTreeError("No pairings to make donations between") 

1344 

1345 done = False 

1346 while not done: 

1347 

1348 # since the pair_values are sorted first by the total diff 

1349 # and then the donation size, the first one that has a 

1350 # positive donation is the best donation 

1351 if donation > 0: 

1352 best_pair = pairing 

1353 best_donation_amount = donation 

1354 done = True 

1355 

1356 # try to get the next pairing if there is one 

1357 try: 

1358 diff, donation, pairing = next(pair_iter) 

1359 except StopIteration: 

1360 # we are done, use the last pairing we had as the 

1361 # pair, its just as good as any of the others let the 

1362 # calling method decide what to do in this situation 

1363 best_pair = pairing 

1364 break 

1365 

1366 

1367 # now we have the best donation and the pair is already in the 

1368 # donor, acceptor order from when we created it 

1369 donor_node_id, acceptor_node_id = best_pair 

1370 

1371 return donor_node_id, acceptor_node_id, best_donation_amount

1374 def _find_best_donation_pair(self, children_donatable_shares, 

1375 children_receivable_shares): 

1376 """This method just returns which children have the most and least 

1377 number of 'shares' which are the effective number of walker 

1378 slots it will be granted in the next segment of dynamics in 

1379 the simulation. This is essentially the amount of sampling 

1380 effort that will be allocated to this region. 

1381  

1382 This method is give the dictionary of the childrens 

1383 

1384 Parameters 

1385 ---------- 

1386 children_donatable_shares : 

1387  

1388 children_receivable_shares : 

1389  

1390 

1391 Returns 

1392 ------- 

1393 

1394 """ 

1395 

1396 # one region will be the donor of shares 

1397 donor_child_node_id = None 

1398 

1399 # the other will accept them 

1400 acceptor_child_node_id = None 

1401 

1402 # record for maximum number of donateable shares 

1403 max_donatable_shares = None 

1404 

1405 # records for the max and min number of shares of the acceptor 

1406 # and donor regions 

1407 donor_n_shares = None 

1408 acceptor_n_shares = None 

1409 

1410 # go through every walker and test it to see if it is either 

1411 # the highest or lowest, record it if it is 

1412 for child_node_id in children_donatable_shares.keys(): 

1413 

1414 # the number of donatable shares is equal to the number 

1415 # of squashable walkers 

1416 n_donatable_shares = children_donatable_shares[child_node_id] 

1417 

1418 # the number of possible shares this node can receive is 

1419 # equal to the number of possible clones it can make 

1420 n_receivable_shares = children_receivable_shares[child_node_id] 

1421 

1422 # we see if this node region is the max region by testing 

1423 # if it is the new highest in shares. It must also be able 

1424 # to donate a share by having at least 1 squashable walker 

1425 if ((donor_child_node_id is None) or (n_shares > donor_n_shares)) and \ 

1426 (n_donatable_shares > 0): 

1427 

1428 # this is a new record 

1429 max_donatable_shares = n_donatable_shares 

1430 

1431 # save how many shares this region has in total 

1432 donor_n_shares = n_shares 

1433 donor_child_node_id = child_node_id 

1434 

1435 

1436 # test if this is the region with the lowest number of 

1437 # shares that is still able to receive at least one share 

1438 if ((acceptor_child_node_id is None) or (n_shares < acceptor_n_shares)) and \ 

1439 (n_receivable_shares > 0): 

1440 

1441 acceptor_n_shares = n_shares 

1442 acceptor_child_node_id = child_node_id 

1443 

1444 # check that both a donor and acceptor were identified and 

1445 # that values for there shares were given 

1446 assert all([True if val is not None else False 

1447 for val in [donor_n_shares, acceptor_n_shares, 

1448 donor_child_node_id, acceptor_child_node_id]]), \ 

1449 "A donor or acceptor was not found" 

1450 

1451 # if the acceptor's number of shares is not less then the 

1452 # donor then there is not possible donation 

1453 if acceptor_n_shares >= donor_n_shares: 

1454 return False 

1455 # if there is a net donation we return the donor and acceptor 

1456 else: 

1457 return donor_child_node_id, acceptor_child_node_id

1459 def _calc_share_donation(self, donor_n_shares, acceptor_n_shares, 

1460 donor_donatable_shares, acceptor_receivable_shares): 

1461 """ 

1462 

1463 Parameters 

1464 ---------- 

1465 donor_n_shares : 

1466  

1467 acceptor_n_shares : 

1468  

1469 donor_donatable_shares : 

1470  

1471 acceptor_receivable_shares : 

1472  

1473 

1474 Returns 

1475 ------- 

1476 

1477 """ 

1478 

1479 # the sibling with the greater number of shares (from both 

1480 # previous resamplings and inherited from the parent) will 

1481 # give shares to the sibling with the least. 

1482

1484 # donation that will make them the most similar, rounding down 

1485 # (i.e. midpoint) 

1486 desired_donation = math.floor((donor_n_shares - acceptor_n_shares)/2)

1489 # and the acceptor has a certain capacity of receiving. Out of 

1490 # the three we can only actually donate the smallest amount 

1491 actual_donation = min(desired_donation, 

1492 donor_donatable_shares, 

1493 acceptor_receivable_shares)

1519 leaf_balance = self.node[leaf]['balance']

1525 leaf_walker_idxs = self.node[leaf]['walker_idxs'] 

1526 leaf_walker_weights = [self.walker_weights[walker_idx] for walker_idx in leaf_walker_idxs]

1528 

1529 # now that we have the walkers that may potentially be merged 

1530 # we need to actually find a set of groupings that satisfies 

1531 # the reduction in balance without any individual walker 

1532 # exceeding the maximum weight. In general this is a difficult 

1533 # problem both here and in deciding how balances are 

1534 # distributed (because the potential merges determine a leafs

1547 # becoming solved it is functionalized here to make a set of

1552 # merge groups" include all the walkers that will be merged 

1553 # and the sum of their weights will be the weight of the final 

1554 # merged walker and should satisfy the maximum weight 

1555 # requirement, i.e. it will not be checked here. 

1556 full_merge_groups_leaf_walker_idxs = \ 

1557 self._solve_merge_groupings(leaf_walker_weights, leaf_balance)

1560 # the "merge groups". Pardon the terminology, but the 

1561 # distinction is trivial and is only relevant to the 

1562 # implementation. The "merge groups" are what is returned. To 

1563 # make them we just choose which walker to keep and which 

1564 # walkers to squash in each full merge group 

1565 for full_merge_group_leaf_walker_idxs in full_merge_groups_leaf_walker_idxs:

1568 # given to the method so we translate them back to the actual 

1569 # walker indices 

1570 chosen_walker_idxs = [leaf_walker_idxs[leaf_walker_idx]

1577 # in the Decision) based on their weights 

1578 

1579 # normalize weights to the sum of all the chosen weights 

1580 chosen_pdist = np.array(chosen_weights) / sum(chosen_weights) 

1581 

1582 # then choose one of the the walker idxs to keep according to 

1583 # their normalized weights 

1584 keep_walker_idx = np.random.choice(chosen_walker_idxs, 1, p=chosen_pdist)[0] 

1585 

1586 # pop the keep idx from the walkers so we can use the rest of 

1587 # them as the squash idxs 

1588 chosen_walker_idxs.pop(chosen_walker_idxs.index(keep_walker_idx)) 

1589 

1590 # the rest are squash_idxs 

1591 squash_walker_idxs = chosen_walker_idxs 

1592 

1593 # update the merge group based on this decision 

1594 merge_groups[keep_walker_idx].extend(squash_walker_idxs) 

1595 

1596 return merge_groups

1642 def _decide_clone_leaf(self, leaf, merge_groups, walkers_num_clones): 

1643 """ 

1644 

1645 Parameters 

1646 ---------- 

1647 leaf : 

1648  

1649 merge_groups : 

1650  

1651 walkers_num_clones : 

1652  

1653 

1654 Returns 

1655 ------- 

1656 

1657 """ 

1658 

1659 # just follow the instructions in the walkers_num_clones and 

1660 # find them slots 

1661 

1662 # this assumes that the all squashes have already been 

1663 # specified in the merge group, this is so that we can use 

1664 # unused walker slots. 

1665 

1666 # if this leaf node was assigned a debt we need to merge 

1667 # walkers 

1668 leaf_balance = self.node[leaf]['balance'] 

1669 leaf_walker_idxs = self.node[leaf]['walker_idxs'] 

1670 leaf_walker_weights = {walker_idx : self.walker_weights[walker_idx] 

1671 for walker_idx in self.node[leaf]['walker_idxs']} 

1672 

1673 # calculate the maximum possible number of clones each free walker 

1674 # could produce 

1675 walker_n_possible_clones = {walker_idx : self._calc_max_num_clones(leaf_weight) 

1676 for walker_idx, leaf_weight in leaf_walker_weights.items()} 

1677 

1678 # the sum of the possible clones needs to be greater than or 

1679 # equal to the balance 

1680 if not sum(walker_n_possible_clones.values()) >= leaf_balance: 

1681 raise RegionTreeError("there isn't enough clones possible to pay the balance") 

1682 

1683 # go through the list of free walkers and see which ones have 

1684 # any possible clones and make a list of them 

1685 cloneable_walker_idxs = [walker_idx for walker_idx in leaf_walker_idxs 

1686 if walker_n_possible_clones[walker_idx] > 0] 

1687 

1688 cloneable_walker_weights = [leaf_walker_weights[walker_idx] 

1689 for walker_idx in cloneable_walker_idxs] 

1690 

1691 # if the sum of them is equal to the leaf balance then we 

1692 # don't have to optimally distribute them and we just give them out 

1693 if sum(walker_n_possible_clones.values()) == leaf_balance: 

1694 

1695 for walker_idx in cloneable_walker_idxs: 

1696 walkers_num_clones[walker_idx] = walker_n_possible_clones[walker_idx] 

1697 

1698 # return this without doing all the prioritization 

1699 return walkers_num_clones 

1700 

1701 # otherwise we want to optimally distribute the clones to the 

1702 # walkers such that we split the largest walkers first 

1703 

1704 # to distribute the clones we iteratively choose the walker 

1705 # with the highest weight after a single clone 

1706 # weight/(n_clones +1) where n_clones is the current number of 

1707 # clones assigned to it (plus itself), and then add another 

1708 # clone to it as long as it is within the range of the number 

1709 # of clones it can make 

1710 

1711 # go until the balance is paid off 

1712 clones_left = leaf_balance 

1713 while clones_left > 0: 

1714 

1715 # determine which walkers are still in the running for 

1716 # receiving clones 

1717 still_cloneable_walker_idxs = [] 

1718 still_cloneable_walker_weights = [] 

1719 for walker_idx, weight in zip(cloneable_walker_idxs, cloneable_walker_weights): 

1720 

1721 # if the number of clones is less than its maximum 

1722 # possible add it to the still applicable ones 

1723 if (walkers_num_clones[walker_idx] < walker_n_possible_clones[walker_idx]): 

1724 

1725 still_cloneable_walker_idxs.append(walker_idx) 

1726 still_cloneable_walker_weights.append(weight) 

1727 

1728 # if there is only one applicable walker left give it the 

1729 # rest of the balance and return 

1730 if len(still_cloneable_walker_idxs) == 1: 

1731 walkers_num_clones[still_cloneable_walker_idxs[0]] += clones_left 

1732 clones_left -= clones_left 

1733 

1734 # end this loop iteration, skipping the decision part 

1735 continue 

1736 

1737 # if there are multiple walkers left we decide between them 

1738 

1739 # calculate the weights of the walker's children given the 

1740 # current number of clones, if num_clones is 0 then it is 

1741 # just it's own weight 

1742 child_weights = [] 

1743 for walker_idx, weight in zip(still_cloneable_walker_idxs, still_cloneable_walker_weights): 

1744 

1745 # the weight of its children given the number of 

1746 # clones already assigned to it 

1747 child_weight = weight / (walkers_num_clones[walker_idx]+1) 

1748 

1749 child_weights.append(child_weight) 

1750 

1751 # get the walker_idx with the highest would-be child weight 

1752 chosen_walker_idx = still_cloneable_walker_idxs[np.argsort(child_weights)[-1]] 

1753 

1754 # add a clone to it 

1755 walkers_num_clones[chosen_walker_idx] += 1 

1756 

1757 # we are one step closer to satisfying the cloning 

1758 # requirement 

1759 clones_left -= 1 

1760 

1761 return walkers_num_clones

1763 def _decide_settle_balance(self): 

1764 """Given the balances of all the leaves figure out actually how to 

1765 settle all the balances. Returns the merge_groups and 

1766 walkers_num_clones 

1767 

1768 Parameters 

1769 ---------- 

1770 

1771 Returns 

1772 ------- 

1773 

1774 """ 

1775 

1776 # initialize the main data structures for specifying how to 

1777 # merge and clone a set of walkers. These will be modified for 

1778 # clones and merges but in the initialized state they will 

1779 # generate all NOTHING records. 

1780 

1781 # the merge groups, a list of lists where the elements of the 

1782 # outer list are individual "merge groups" that themselves 

1783 # contain the elements of the walkers that will be squashed in 

1784 # a merge and the index of the merge group in the outer list 

1785 # is the index of the walker that will be kept 

1786 # (i.e. KEEP_MERGE and have its state persist to the next 

1787 # step). Indices appearing in any merge group can themselves 

1788 # not have a merge group 

1789 merge_groups = [[] for i in self.walker_weights] 

1790 

1791 # the number of clones to make for each walker. Simply a list 

1792 # of 0 or positive integers that specify how many EXTRA clones 

1793 # will be made. E.g. if a cloned walker is to have 3 children 

1794 # then the number of clones is 2. We consider clones copies 

1795 # from the original walker which is given by the index in the 

1796 # list. This number then gives the number of new slots needed 

1797 # for a cloning event 

1798 walkers_num_clones = [0 for i in self.walker_weights] 

1799 

1800 # get all the leaf balances 

1801 leaf_nodes = self.leaf_nodes() 

1802 leaf_balances = [self.node[leaf]['balance'] for leaf in leaf_nodes] 

1803 

1804 # get the negative and positive balanced leaves 

1805 neg_leaves = [leaf_nodes[leaf_idx[0]] for leaf_idx in 

1806 np.argwhere(np.array(leaf_balances) < 0)] 

1807 pos_leaves = [leaf_nodes[leaf_idx[0]] for leaf_idx in 

1808 np.argwhere(np.array(leaf_balances) > 0)] 

1809 

1810 # we decide on how the walkers will be cloned and 

1811 # merged. These steps are purely functional and do not modify 

1812 # any attributes on the RegionTree. The merge_groups and 

1813 # walkers_num_clones can be used to commit these changes 

1814 # elsewhere if desired. 

1815 

1816 # first do all leaves with negative balances, so that after we 

1817 # have freed up slots we can fill them with clones since in 

1818 # WEXplore we want to have an economy of slots and not create 

1819 # them if we don't have to 

1820 for leaf in neg_leaves: 

1821 merge_groups = self._decide_merge_leaf(leaf, merge_groups) 

1822 

1823 # then do all the leaves with positive balances to fill the 

1824 # slots left from squashing walkers 

1825 for leaf in pos_leaves: 

1826 walkers_num_clones = self._decide_clone_leaf(leaf, merge_groups, walkers_num_clones) 

1827 

1828 

1829 return merge_groups, walkers_num_clones

1965 if len(underweight_walker_idxs > 0): 

1966 

1967 # list of clone parents that will produce underweight 

1968 # walkers 

1969 underweight_producer_idxs = []

1984 def balance_tree(self, delta_walkers=0): 

1985 """Do balancing between the branches of the tree. the `delta_walkers` 

1986 kwarg can be used to increase or decrease the total number of 

1987 walkers, but defaults to zero which will cause no net change 

1988 in the number of walkers. 

1989 

1990 Parameters 

1991 ---------- 

1992 delta_walkers : 

1993 (Default value = 0) 

1994 

1995 Returns 

1996 ------- 

1997 

1998 """ 

1999 

2000 # set the delta walkers to the balance of the root node 

2001 self.node[self.ROOT_NODE]['balance'] = delta_walkers 

2002 

2003 # do a breadth first traversal and balance at each level 

2004 for parent, children in nx.bfs_successors(self, self.ROOT_NODE): 

2005 

2006 # pass on the balance of this parent to the children from the 

2007 # parents, distribute walkers between 

2008 parental_balance = self.node[parent]['balance'] 

2009 

2010 # this will both propagate the balance set for the root 

2011 # walker down the tree and balance between the children 

2012 self._propagate_and_balance_shares(parental_balance, children) 

2013 

2014 # check that the sum of the balances of the leaf nodes 

2015 # balances to delta_walkers 

2016 leaf_balances = [self.node[leaf]['balance'] for leaf in self.leaf_nodes()] 

2017 if sum(leaf_balances) != delta_walkers: 

2018 

2019 raise RegionTreeError( 

2020 "The balances of the leaf nodes ({}) do not balance to delta_walkers ({})".format( 

2021 leaf_balances, delta_walkers)) 

2022 

2023 # decide on how to settle all the balances between leaves 

2024 merge_groups, walkers_num_clones = self._decide_settle_balance() 

2025 

2026 # count up the number of clones and merges in the merge_groups 

2027 # and the walkers_num_clones 

2028 num_clones = sum(walkers_num_clones) 

2029 num_squashed = sum([len(merge_group) for merge_group in merge_groups]) 

2030 

2031 # check that the number of clones and number of squashed 

2032 # walkers balance to the delta_walkers amount 

2033 if num_clones - num_squashed != delta_walkers: 

2034 

2035 raise RegionTreeError("The number of new clones ({}) is not balanced by the number of" 

2036 "squashed walkers ({}) to the delta_walkers specified ({})".format( 

2037 num_clones, num_squashed, delta_walkers)) 

2038 

2039 # DEBUG 

2040 # check the merge groups and walkers_num_clones to make sure 

2041 # they are valid 

2042 try: 

2043 self._check_clone_merge_specs(merge_groups, walkers_num_clones) 

2044 except ResamplerError as resampler_err: 

2045 print(resampler_err) 

2046 import ipdb; ipdb.set_trace() 

2047 

2048 return merge_groups, walkers_num_clones

2050class WExploreResampler(CloneMergeResampler): 

2051 """Resampler implementing the WExplore algorithm. 

2052 

2053 See the paper for a full description of the algorithm, but 

2054 briefly: 

2055 

2056 WExplore defines a hierarchical Voronoi tesselation on a subspace 

2057 of the full walker state. Regions in a Voronoi cell are defined by 

2058 a point in this subspace called an 'image', and are scoped by 

2059 their enclosing region in the hierarchy. 

2060 

2061 The hierarchy is defined initially by: 

2062 

2063 - number of levels of the hierarchy, AKA depth 

2064 - number of regions allowed at a level of the hierarchy 

2065 - a cutoff 'distance' at each level of the hierarchy 

2066 

2067 All regions have a unique specification (called a leaf_id) which 

2068 is a k-tuple of region indices, where k is the depth of the 

2069 hierarchy. 

2070 

2071 At first the hierarchy only has a single region at all levels 

2072 which is given by the image of the 'init_state' constructor 

2073 argument. 

2074 

2075 For a hierarchy of depth 4 the leaf_id for this first region is 

2076 (0, 0, 0, 0) which indicates that at level 0 (the highest level) 

2077 we are selecting the first region (region 0), and at level 1 we 

2078 select the first region at that level (region 0 again), and so on. 

2079 

2080 For this region all images at each level are identical. 

2081 

2082 During resampling walkers first are binned into the region they 

2083 fall. This is achieved using a breadth first search where the 

2084 distance (according to the distance metric) between the walker 

2085 state and all region images are computed. The image that is 

2086 closest to the walker is selected and in the next iteration of 

2087 comparisons we restrict the search to only regions within this 

2088 super-region. 

2089 

2090 For example for a region tree with the following leaf ids: (0,0,0) 

2091 and (0,1,0) we skip distance computations at level 0, since there 

2092 is only one region. At level 1, we compute the distance to images 

2093 0 and 1 and choose the one that is closest. Since each of these 

2094 regions only has 1 sub-region we do not need to recalculate the 

2095 distance and can assign the walker. 

2096 

2097 Indeed every super-region will have exactly one sub-region that 

2098 has the same image as it, and these distance calculations are 

2099 never repeated for performance.

2101 This resampler adds the additional 'region_assignment' field to 

2102 the resampling records which indicates the region a walker was 

2103 assigned to during resampling.

2107 specification of the cutoff distances at each level indicate when 

2108 sufficient novelty in a walker motivates creation of a new region.

2111 is saved for each level, e.g. (0.1, 0.3, 0.5) for the example 

2112 above. This is compared to the cutoff distance specification, 

2113 e.g. (0.2, 0.2, 0.2). The highest level at which a distance

2118 takes precedence over the lower level.

2122 the branch occurred.

2126 the region created, which would be (0, 2, 0) for this example.

2129 triggered the branching.

2132 change during branching.

2135 sub-regions for a super-region which is given for each level. For 

2136 example, (5, 10, 10) indicates that regions at the top-most level 

2137 can have 5 sub-regions, and in turn those sub-regions can have 10 

2138 sub-regions, and so on.

2141 regions and includes: 

2142 

2143 - branching_level : the level of the tree the branching occured 

2144 at, which relates it to the number of allowed regions and the

2158 That covers how regions are initialized, adaptively created, and 

2159 recorded, but doesn't explain how these regions are used inform 

2160 the actual resampling process. 

2161 

2162 Essentially, the hierarchical structure allows for balanced 

2163 resource trading between regions. The resource in the case of 

2164 weighted ensemble is the allocation of a walker which will be 

2165 scheduled for sampling in the next cycle. The more walkers a 

2166 region has the more sampling of the region will occur. 

2167 

2168 After the sampling step walkers move around in regions and between 

2169 regions. Some regions will end up collecting more walkers than 

2170 other regions during this. The goal is to redistribute those 

2171 walkers to other regions so that each level of the hierarchy is as 

2172 balanced as possible. 

2173 

2174 So after sampling the number of walkers is added up for each 

2175 region. At the top of the hierarchy a collection of trades between 

2176 the super-regions is negotiated based on which regions are able to 

2177 give up walkers and those that need them such that each region has 

2178 as close to the same number of walkers as possible. 

2179 

2180 The 'payment' of a tax (or donation) is performed by merging 

2181 walkers and the reception of the donation (or welfare 

2182 dispensation) is achieved by cloning walkers. 

2183 

2184 Because there are constraints on how many walkers can be merged 

2185 and cloned due to minimum and maximum walker probabilities and 

2186 total number of walkers, sometimes a region may have an excess of 

2187 walkers but none of them are taxable (donatable) because any merge 

2188 would create a walker above the maximum probability. Conversely, a 

2189 region may not be able to receive walkers because any clone of a 

2190 walker would make walkers with weight lower than the minimum. 

2191 

2192 However, super-regions cannot actually 'pay' for these trades 

2193 themselves and simply make a request to their sub-regions to 

2194 provide the requested walkers (or to find room for accepting 

2195 them). The trade negotiation process then repeats within each 

2196 sub-region. When the request for debits and credits finally 

2197 reaches leaf node regions the process stops and the actual 

2198 identities of the walkers that will be cloned or merged are 

2199 determined. 

2200 

2201 Then the cloning and merging of walkers is performed. 

2202 

2203 After this process if we were to recount the number of walkers in 

2204 all regions then they should be pretty well balanced at each 

2205 level. 

2206 

2207 Its worth noting that some regions will have no walkers because 

2208 once a region contains no walkers (all walkers leave the region 

2209 during sampling) it can no longer receive any walkers at all 

2210 because it cannot clone. Of course walkers may re-enter the region 

2211 and repopulate it but until that happens these regions are 

2212 excluded from negotiations. 

2213 

2214 Only the net clones and merges are recorded in the records.
