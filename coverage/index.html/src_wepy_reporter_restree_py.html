216 def _make_record(record_d, cycle_idx, 

217 record_field_names, field_names, field_shapes, rec_namedtuple): 

218 """Generic record making function. 

219 

220 Parameters 

221 ---------- 

222 record_d : dict of str : value 

223 

224 cycle_idx : int 

225 

226 record_field_names : list of str 

227 The record field names to use for the record. 

228 

229 field_names : list of str 

230 All of the field names available for the field type. 

231 

232 field_shapes : list of tuple of int 

233 Shapes for each field 

234 

235 rec_namedtuple : namedtuple object 

236 The namedtuple class to make records out of 

237 

238 Returns 

239 ------- 

240 

241 record : namedtuple object 

242 

243 """ 

244 

245 # go through each field of the record 

246 rec_d = {'cycle_idx' : cycle_idx} 

247 for field_name, field_value in record_d.items(): 

248 

249 field_idx = field_names.index(field_name) 

250 field_shape = field_shapes[field_idx] 

251 

252 # if it is not part of the record we just keep going and 

253 # don't bother 

254 if field_name not in record_field_names: 

255 continue 

256 

257 # otherwise we need to do some formatting 

258 else: 

259 

260 # if it is variable length (shape is Ellipsis) or if 

261 # it has more than one element cast all elements to 

262 # tuples 

263 if field_shape is Ellipsis: 

264 value = tuple(field_value) 

265 

266 # if it is not variable length make sure it is not more than a 

267 # 1D feature vector 

268 elif len(field_shape) > 2: 

269 raise TypeError( 

270 "cannot convert fields with feature vectors more than 1 dimension," 

271 " was given {} for {}".format( 

272 field_value.shape[1:], field_name)) 

273 

274 # if it is only a rank 1 feature vector and it has more than 

275 # one element make a tuple out of it 

276 elif field_shape[0] > 1: 

277 value = tuple(field_value) 

278 

279 # otherwise just get the single value instead of keeping it as 

280 # a single valued feature vector 

281 else: 

282 value = field_value[0] 

283 

284 rec_d[field_name] = value 

285 

286 # after you make the dictionary of this convert it to a 

287 # namedtuple record 

288 record = rec_namedtuple(*(rec_d[key] for key in rec_namedtuple._fields)) 

289 

290 return record

293 def report(self, cycle_idx=None, 

294 resampled_walkers=None, 

295 warp_data=None, 

296 progress_data=None, 

297 resampling_data=None, 

298 **kwargs): 

299 """Generate the resampling tree GEXF file. 

300 

301 Parameters 

302 ---------- 

303 cycle_idx : int 

304 

305 resampled_walkers : list of Walker 

306 

307 warp_data : dict of str : value 

308 

309 progress_data : dict of str : value 

310 

311 resampling_data : dict of str : value 

312 

313 """ 

314 

315 

316 # we basically want to generate a new parent table from the 

317 # records we were just given and then add that to our existing 

318 # parent table 

319 

320 # we first have to generate the resampling and warping records 

321 # though adding the cycle_idx 

322 

323 resampling_records = [self._make_resampling_record(rec_d, cycle_idx) 

324 for rec_d in resampling_data] 

325 

326 warping_records = [self._make_warping_record(rec_d, cycle_idx) 

327 for rec_d in warp_data] 

328 

329 # tabulate the discontinuities 

330 for warping_record in warping_records: 

331 # if this record classifies as discontinuous 

332 if self._bc.warping_discontinuity(warping_record): 

333 # then we save it as one of the nodes that is discontinuous 

334 disc_node_id = (warping_record.cycle_idx, warping_record.walker_idx) 

335 self._discontinuous_nodes.append(disc_node_id) 

336 

337 # get the weights of the resampled walkers since we will want 

338 # to plot them 

339 walker_weights = [walker.weight for walker in resampled_walkers] 

340 # add these to the collected weights 

341 self._walker_weights.append(walker_weights) 

342 

343 # get the progress values for all of the walkers 

344 walkers_progress = [progress for progress in progress_data[self.progress_key]] 

345 self._walkers_progress.append(walkers_progress) 

346 

347 # so we make a resampling panel from the records 

348 

349 # TODO: this uses the cycle_idx in the records and thus wehn 

350 # you receive just one cycle it adds a empty records to the 

351 # beginning of it and only the last entry is the one we are 

352 # looking for so for now we just get it, and wrap back in a list 

353 res_panel = [resampling_panel(resampling_records, is_sorted=False)[-1]] 

354 

355 # then the parent panel, and then the net parent table 

356 par_panel = parent_panel(self._decision_class, res_panel) 

357 cycle_parent_table = net_parent_table(par_panel) 

358 

359 # add these to the main parent table 

360 self._parent_table.extend(cycle_parent_table) 

361 

362 # now use this to create a new ParentForest without a WepyHDF5 

363 # file view. We exclude the discontinuities on purpose because 

364 # we want to keep the paren child relationships through the 

365 # warps for visualization and will just change the shapes of 

366 # the nodes to communicate the warping 

367 parent_forest = ParentForest(parent_table=self._parent_table) 

368 

369 # The only ones that should be none are the roots which will 

370 # be 1.0 / N 

371 n_roots = len(self._parent_table[0]) 

372 root_weight = 1.0 / n_roots 

373 

374 # compute the size of the nodes to plot, which is the 

375 # free_energy. To compute this we flatten the list of list of 

376 # weights to a 1-d array then unravel it back out 

377 

378 # put the root weights at the beginning of this array, and 

379 # account for it in the flattening 

380 flattened_weights = [root_weight for _ in self._walker_weights[0]] 

381 cycles_n_walkers = [len(self._walker_weights[0])] 

382 for cycle_weights in self._walker_weights: 

383 cycles_n_walkers.append(len(cycle_weights)) 

384 flattened_weights.extend(cycle_weights) 

385 

386 # now compute the free energy 

387 flattened_free_energies = free_energy(np.array(flattened_weights)) 

388 

389 # and ravel them back into a list of lists 

390 free_energies = [] 

391 last_index = 0 

392 for cycle_n_walkers in cycles_n_walkers: 

393 free_energies.append(flattened_free_energies[last_index:last_index + cycle_n_walkers]) 

394 last_index += cycle_n_walkers 

395 

396 # we want to set the colors based on the progress, first we 

397 # scale all the progress values to a 1-100 scale 

398 if self.max_progress_value is not None: 

399 norm_ratio = self.MAX_PROGRESS_NORM / self.max_progress_value 

400 

401 # otherwise we use the maximum value from the full progress 

402 # values dataset so far 

403 else: 

404 max_progress_value = max([max(row) for row in self._walkers_progress]) 

405 # then use that for the ratio 

406 norm_ratio = self.MAX_PROGRESS_NORM / max_progress_value 

407 

408 # now that we have ratios for normalizing values we apply that 

409 # and then use the lookup table for the color bar to get the 

410 # RGB color values 

411 colors = [] 

412 for progress_row in self._walkers_progress: 

413 color_row = [self.colormap(progress * norm_ratio, bytes=True) 

414 for progress in progress_row] 

415 colors.append(color_row) 

416 

417 # put these into the parent forest graph as node attributes, 

418 # so we can get them out as a lookup table by node id for 

419 # assigning to nodes in the layout. We skip the first row of 

420 # free energies for the roots in this step 

421 parent_forest.set_attrs_by_array('free_energy', free_energies[1:]) 

422 parent_forest.set_attrs_by_array('color', colors) 

423 

424 # get the free energies out as a dictionary, flattening it to 

425 # a single value. Set the nodes with None for their FE to a 

426 # small number. 

427 

428 # now we make the graph which will contain the layout 

429 # information 

430 layout_forest = LayoutGraph(parent_forest.graph) 

431 

432 

433 # get the free energy attributes out using the root weight for 

434 # the nodes with no fe defined 

435 

436 # start it with the root weights 

437 node_fes = {(-1, i) : free_energy for i, free_energy in enumerate(free_energies[0])} 

438 for node_id, fe_arr in layout_forest.get_node_attributes('free_energy').items(): 

439 

440 if fe_arr is None: 

441 node_fes[node_id] = 0.0 

442 else: 

443 node_fes[node_id] = fe_arr 

444 

445 # the default progress for the root ones is 0 and so the color 

446 # is also 0 

447 node_colors = {} 

448 for node_id, color_arr in layout_forest.get_node_attributes('color').items(): 

449 

450 if color_arr is None: 

451 # make a black color for these 

452 node_colors[node_id] = tuple(int(255) for a in range(4)) 

453 else: 

454 node_colors[node_id] = color_arr 

455 

456 # make a dictionary for the shapes of the nodes 

457 node_shapes = {} 

458 # set all the nodes to a default 'disc' 

459 for node in layout_forest.viz_graph.nodes: 

460 node_shapes[node] = self.DEFAULT_NODE_SHAPE 

461 # then set all the discontinuous ones 

462 for discontinuous_node in self._discontinuous_nodes: 

463 node_shapes[discontinuous_node] = self.DISCONTINUOUS_NODE_SHAPE 

464 

465 # also set the color to black 

466 for discontinuous_node in self._discontinuous_nodes: 

467 node_colors[discontinuous_node] = tuple(int(0) for a in range(4)) 

468 

469 

470 # we are going to output to the gexf format so we use the 

471 # pertinent methods 

472 

473 # we set the sizes 

474 layout_forest.set_node_gexf_sizes(node_fes) 

475 

476 # and set the colors based on the progresses 

477 

478 layout_forest.set_node_gexf_colors_rgba(node_colors) 

479 

480 # then set the shape of the nodes based on whether they were 

481 # warped or not 

482 layout_forest.set_node_gexf_shape(node_shapes) 

483 

484 

485 # now we get to the part where we make the layout (positioning 

486 # of nodes), so we parametrize a layout engine 

487 tree_layout = ResamplingTreeLayout(row_spacing=self.row_spacing, 

488 step_spacing=self.step_spacing, 

489 node_radius=self.default_node_radius) 

490 

491 

492 node_coords = tree_layout.layout(parent_forest, 

493 node_radii=node_fes) 

494 

495 layout_forest.set_node_gexf_positions(node_coords) 

496 

497 

498 # now we can write this graph to the gexf format to the file 

499 nx.write_gexf(layout_forest.viz_graph, self.gexf_restree_path)
